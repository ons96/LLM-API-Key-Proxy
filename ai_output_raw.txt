import logging
import time
from typing import Any, Dict, Optional, List

from fastapi import APIRouter, HTTPException, Request
from pydantic import BaseModel, Field

# Attempt to import unified model registry if available
try:
    from ..rotator_library.model_info_service import model_registry
    HAS_MODEL_REGISTRY = True
except (ImportError, AttributeError):
    HAS_MODEL_REGISTRY = False
    model_registry = None

from ..rotator_library.model_definitions import ModelDefinitions

logger = logging.getLogger(__name__)

router = APIRouter()


class ModelPermission(BaseModel):
    """OpenAI-compatible model permission object."""
    id: str = "modelperm"
    object: str = "model_permission"
    created: int = Field(default_factory=lambda: int(time.time()))
    allow_create_engine: bool = False
    allow_sampling: bool = True
    allow_logprobs: bool = True
    allow_search_indices: bool = False
    allow_view: bool = True
    allow_fine_tuning: bool = False
    organization: str = "*"
    group: Optional[str] = None
    is_blocking: bool = False


class ModelResponse(BaseModel):
    """OpenAI-compatible model response object."""
    id: str
    object: str = "model"
    created: int
    owned_by: str
    permission: List[ModelPermission]
    root: Optional[str] = None
    parent: Optional[str] = None


def _find_model_in_registry(model_id: str) -> Optional[Dict[str, Any]]:
    """Search for model in the unified model registry (model_info_service)."""
    if not HAS_MODEL_REGISTRY or not model_registry:
        return None
    
    try:
        # Try exact match first
        if hasattr(model_registry, 'get_model'):
            model = model_registry.get_model(model_id)
            if model:
                return {
                    "id": model.model_id,
                    "owned_by": model.provider or "unknown",
                    "created": model.timestamp if model.timestamp else int(time.time()),
                    "root": model.model_id,
                }
        
        # Try finding by alias or fuzzy match
        if hasattr(model_registry, 'find_model'):
            model = model_registry.find_model(model_id)
            if model:
                return {
                    "id": model.model_id,
                    "owned_by": model.provider or "unknown",
                    "created": model.timestamp if model.timestamp else int(time.time()),
                    "root": model.model_id,
                }
    except Exception as e:
        logger.debug(f"Registry lookup failed for {model_id}: {e}")
    
    return None


def _find_model_in_definitions(model_id: str) -> Optional[Dict[str, Any]]:
    """Search for model in environment-based definitions."""
    model_defs = ModelDefinitions()
    
    # Handle provider/model format (e.g., "openai/gpt-4")
    if "/" in model_id:
        parts = model_id.split("/", 1)
        if len(parts) == 2:
            provider, name = parts
            definition = model_defs.get_model_definition(provider, name)
            if definition:
                return {
                    "id": model_id,
                    "owned_by": provider,
                    "created": int(time.time()),
                    "root": model_id,
                }
    else:
        # Search across all providers for the model name
        for provider in model_defs.definitions.keys():
            definition = model_defs.get_model_definition(provider, model_id)
            if definition:
                full_id = f"{provider}/{model_id}"
                return {
                    "id": full_id,
                    "owned_by": provider,
                    "created": int(time.time()),
                    "root": full_id,
                }
    
    return None


@router.get("/v1/models/{model_id}", response_model=ModelResponse)
async def get_model(model_id: str, request: Request):
    """
    Retrieve a specific model by ID.
    
    Returns model metadata in OpenAI-compatible format.
    Checks unified registry first (if available), then falls back to provider definitions.
    
    Args:
        model_id: The model identifier (e.g., "gpt-4", "openai/gpt-4")
        
    Returns:
        ModelResponse with model details
        
    Raises:
        HTTPException: 404 if model not found
    """
    logger.debug(f"Model info requested: {model_id}")
    
    # Try unified registry first for rich metadata
    model_data = _find_model_in_registry(model_id)
    
    # Fall back to environment definitions
    if not model_data:
        model_data = _find_model_in_definitions(model_id)
    
    if not model_data:
        raise HTTPException(
            status_code=404,
            detail={
                "error": {
                    "message": f"The model '{model_id}' does not exist or you do not have access to it.",
                    "type": "invalid_request_error",
                    "param": "model",
                    "code": "model_not_found"
                }
            }
        )
    
    return ModelResponse(
        id=model_data["id"],
        object="model",
        created=model_data.get("created", int(time.time())),
        owned_by=model_data.get("owned_by", "unknown"),
        permission=[ModelPermission()],
        root=model_data.get("root", model_id),
        parent=None
    )
