# src/rotator_library/providers/provider_cache.py
"""
Shared cache utility for providers.

A modular, async-capable cache system supporting:
- Dual-TTL: short-lived memory cache, longer-lived disk persistence
- Background persistence with batched writes
- Automatic cleanup of expired entries
- Cache invalidation strategies (key, pattern, tag, age-based)
- LRU eviction for memory pressure management
- Generic key-value storage for any provider-specific needs

Usage examples:
- Gemini 3: thoughtSignatures (tool_call_id → encrypted signature)
- Claude: Thinking content (composite_key → thinking text + signature)
- General: Any transient data that benefits from persistence across requests
"""

from __future__ import annotations

import asyncio
import json
import logging
import os
import re
import time
from pathlib import Path
from typing import Any, Dict, List, Optional, Set, Tuple, Union

from ..utils.resilient_io import safe_write_json

lib_logger = logging.getLogger("rotator_library")


# =============================================================================
# UTILITY FUNCTIONS
# =============================================================================


def _env_bool(key: str, default: bool = False) -> bool:
    """Get boolean from environment variable."""
    return os.getenv(key, str(default).lower()).lower() in ("true", "1", "yes")


def _env_int(key: str, default: int) -> int:
    """Get integer from environment variable."""
    return int(os.getenv(key, str(default)))


# =============================================================================
# CACHE ENTRY
# =============================================================================


class CacheEntry:
    """Represents a cached value with metadata."""

    def __init__(
        self,
        value: str,
        timestamp: float,
        tags: Optional[Set[str]] = None,
        access_count: int = 0,
        last_accessed: Optional[float] = None,
    ):
        self.value = value
        self.timestamp = timestamp
        self.tags = tags or set()
        self.access_count = access_count
        self.last_accessed = last_accessed or timestamp

    def to_dict(self) -> Dict[str, Any]:
        """Serialize to dictionary for disk storage."""
        return {
            "value": self.value,
            "timestamp": self.timestamp,
            "tags": list(self.tags),
            "access_count": self.access_count,
            "last_accessed": self.last_accessed,
        }

    @classmethod
    def from_dict(cls, data: Dict[str, Any]) -> "CacheEntry":
        """Deserialize from dictionary."""
        return cls(
            value=data.get("value", ""),
            timestamp=data.get("timestamp", 0),
            tags=set(data.get("tags", [])),
            access_count=data.get("access_count", 0),
            last_accessed=data.get("last_accessed", data.get("timestamp", 0)),
        )


# =============================================================================
# PROVIDER CACHE CLASS
# =============================================================================


class ProviderCache:
    """
    Server-side cache for provider conversation state preservation.

    A generic, modular cache supporting any key-value data that providers need
    to persist across requests. Features:

    - Dual-TTL system: entries live in memory for memory_ttl, but persist on
      disk for the longer disk_ttl. Memory cleanup does NOT affect disk entries.
    - Merge-on-save: disk writes merge current memory with existing disk entries,
      preserving disk-only entries until they exceed disk_ttl
    - Async disk persistence with batched writes
    - Background cleanup task for memory-expired entries (disk untouched)
    - Cache invalidation: key-based, pattern-based, tag-based, age-based
    - LRU eviction: automatic removal of least recently used entries when
      memory limits are exceeded
    - Statistics tracking (hits, misses, writes, disk preservation, invalidations)

    Args:
        cache_file: Path to disk cache file
        memory_ttl_seconds: In-memory entry lifetime (default: 1 hour)
        disk_ttl_seconds: Disk entry lifetime (default: 48 hours)
        enable_disk: Whether to enable disk persistence (default: from env or True)
        write_interval: Seconds between background disk writes (default: 60)
        cleanup_interval: Seconds between expired entry cleanup (default: 30 min)
        max_memory_entries: Maximum entries in memory before LRU eviction (default: 10000)
        env_prefix: Environment variable prefix for configuration overrides

    Environment Variables (with default prefix "PROVIDER_CACHE"):
        {PREFIX}_ENABLE: Enable/disable disk persistence
        {PREFIX}_WRITE_INTERVAL: Background write interval in seconds
        {PREFIX}_CLEANUP_INTERVAL: Cleanup interval in seconds
        {PREFIX}_MAX_MEMORY_ENTRIES: Maximum memory cache size
    """

    def __init__(
        self,
        cache_file: Path,
        memory_ttl_seconds: int = 3600,
        disk_ttl_seconds: int = 172800,  # 48 hours
        enable_disk: Optional[bool] = None,
        write_interval: Optional[int] = None,
        cleanup_interval: Optional[int] = None,
        max_memory_entries: Optional[int] = None,
        env_prefix: str = "PROVIDER_CACHE",
    ):
        # In-memory cache: {cache_key: CacheEntry}
        self._cache: Dict[str, CacheEntry] = {}
        self._memory_ttl = memory_ttl_seconds
        self._disk_ttl = disk_ttl_seconds
        self._lock = asyncio.Lock()
        self._disk_lock = asyncio.Lock()

        # Tag index for efficient tag-based invalidation: {tag: set(keys)}
        self._tag_index: Dict[str, Set[str]] = {}

        # Disk persistence configuration
        self._cache_file = cache_file
        self._enable_disk = (
            enable_disk
            if enable_disk is not None
            else _env_bool(f"{env_prefix}_ENABLE", True)
        )
        self._dirty = False
        self._write_interval = write_interval or _env_int(
            f"{env_prefix}_WRITE_INTERVAL", 60
        )
        self._cleanup_interval = cleanup_interval or _env_int(
            f"{env_prefix}_CLEANUP_INTERVAL", 1800
        )
        self._max_memory_entries = max_memory_entries or _env_int(
            f"{env_prefix}_MAX_MEMORY_ENTRIES", 10000
        )

        # Background tasks
        self._writer_task: Optional[asyncio.Task] = None
        self._cleanup_task: Optional[asyncio.Task] = None
        self._running = False

        # Statistics
        self._stats = {
            "memory_hits": 0,
            "disk_hits": 0,
            "misses": 0,
            "writes": 0,
            "disk_errors": 0,
            "invalidations": 0,
            "evictions": 0,
            "pattern_invalidations": 0,
            "tag_invalidations": 0,
            "bulk_invalidations": 0,
        }

        # Track disk health for monitoring
        self._disk_available = True

        # Metadata about this cache instance
        self._cache_name = cache_file.stem if cache_file else "unnamed"

        if self._enable_disk:
            lib_logger.debug(
                f"ProviderCache[{self._cache_name}]: Disk enabled "
                f"(memory_ttl={memory_ttl_seconds}s, disk_ttl={disk_ttl_seconds}s, "
                f"max_entries={self._max_memory_entries})"
            )
            asyncio.create_task(self._async_init())
        else:
            lib_logger.debug(f"ProviderCache[{self._cache_name}]: Memory-only mode")

    # =========================================================================
    # INITIALIZATION
    # =========================================================================

    async def _async_init(self) -> None:
        """Async initialization: load from disk and start background tasks."""
        try:
            await self._load_from_disk()
            await self._start_background_tasks()
        except Exception as e:
            lib_logger.error(
                f"ProviderCache[{self._cache_name}] async init failed: {e}"
            )

    async def _load_from_disk(self) -> None:
        """Load cache from disk file with TTL validation."""
        if not self._enable_disk or not self._cache_file.exists():
            return

        try:
            async with self._disk_lock:
                with open(self._cache_file, "r", encoding="utf-8") as f:
                    data = json.load(f)

                if data.get("version") != "2.0":
                    # Migrate from old format if necessary
                    await self._migrate_old_format(data)
                    return

                now = time.time()
                entries = data.get("entries", {})
                loaded = expired = 0

                for cache_key, entry_data in entries.items():
                    age = now - entry_data.get("timestamp", 0)
                    if age <= self._disk_ttl:
                        entry = CacheEntry.from_dict(entry_data)
                        self._cache[cache_key] = entry
                        # Rebuild tag index
                        for tag in entry.tags:
                            self._tag_index.setdefault(tag, set()).add(cache_key)
                        loaded += 1
                    else:
                        expired += 1

                lib_logger.debug(
                    f"ProviderCache[{self._cache_name}]: Loaded {loaded} entries ({expired} expired)"
                )
        except json.JSONDecodeError as e:
            lib_logger.warning(
                f"ProviderCache[{self._cache_name}]: File corrupted: {e}"
            )
        except Exception as e:
            lib_logger.error(f"ProviderCache[{self._cache_name}]: Load failed: {e}")

    async def _migrate_old_format(self, data: Dict[str, Any]) -> None:
        """Migrate from v1.0 format to v2.0."""
        if data.get("version") == "1.0":
            lib_logger.info(f"ProviderCache[{self._cache_name}]: Migrating from v1.0")
            now = time.time()
            entries = data.get("entries", {})
            for cache_key, entry in entries.items():
                age = now - entry.get("timestamp", 0)
                if age <= self._disk_ttl:
                    value = entry.get("value", entry.get("signature", ""))
                    if value:
                        self._cache[cache_key] = CacheEntry(
                            value=value,
                            timestamp=entry["timestamp"],
                        )
            self._dirty = True

    # =========================================================================
    # CORE OPERATIONS
    # =========================================================================

    async def get(self, key: str) -> Optional[str]:
        """
        Retrieve value from cache.

        Checks memory first, then disk (if enabled and memory miss).
        Updates access statistics for LRU tracking.
        """
        async with self._lock:
            now = time.time()
            entry = self._cache.get(key)

            if entry:
                # Check memory TTL
                if now - entry.timestamp <= self._memory_ttl:
                    entry.access_count += 1
                    entry.last_accessed = now
                    self._stats["memory_hits"] += 1
                    return entry.value
                else:
                    # Memory expired but might be on disk
                    pass

            # Try disk if memory miss/expired and disk enabled
            if self._enable_disk and self._disk_available:
                # Note: Disk entries are loaded into memory on startup,
                # so this path is only hit if entry was evicted from memory
                # but is still valid on disk. We don't load individual entries
                # from disk on demand for simplicity.
                pass

            self._stats["misses"] += 1
            return None

    async def set(
        self,
        key: str,
        value: str,
        tags: Optional[Set[str]] = None,
        custom_ttl: Optional[int] = None,
    ) -> None:
        """
        Store value in cache.

        Args:
            key: Cache key
            value: Value to store
            tags: Optional set of tags for group invalidation
            custom_ttl: Optional custom TTL for this entry (overrides default)
        """
        async with self._lock:
            now = time.time()
            
            # Handle existing entry tags cleanup
            if key in self._cache:
                old_entry = self._cache[key]
                for tag in old_entry.tags:
                    if tag in self._tag_index and key in self._tag_index[tag]:
                        self._tag_index[tag].remove(key)
                        if not self._tag_index[tag]:
                            del self._tag_index[tag]

            # Create new entry
            entry = CacheEntry(
                value=value,
                timestamp=now,
                tags=tags or set(),
            )
            
            self._cache[key] = entry
            
            # Update tag index
            if tags:
                for tag in tags:
                    self._tag_index.setdefault(tag, set()).add(key)

            self._dirty = True
            
            # Check for LRU eviction if over limit
            if len(self._cache) > self._max_memory_entries:
                await self._evict_lru_entries(1)

    async def get_and_invalidate(self, key: str) -> Optional[str]:
        """
        Atomically retrieve and remove a value from cache.
        
        Useful for one-time tokens or consumed signatures.
        """
        async with self._lock:
            entry = self._cache.pop(key, None)
            
            if entry:
                # Cleanup tag index
                for tag in entry.tags:
                    if tag in self._tag_index and key in self._tag_index[tag]:
                        self._tag_index[tag].remove(key)
                        if not self._tag_index[tag]:
                            del self._tag_index[tag]
                
                self._stats["invalidations"] += 1
                self._dirty = True
                return entry.value
            
            return None

    # =========================================================================
    # CACHE INVALIDATION STRATEGIES
    # =========================================================================

    async def invalidate(self, key: str) -> bool:
        """
        Invalidate a specific cache entry by key.
        
        Returns:
            True if entry existed and was removed, False otherwise.
        """
        async with self._lock:
            if key in self._cache:
                entry = self._cache.pop(key)
                
                # Cleanup tag index
                for tag in entry.tags:
                    if tag in self._tag_index and key in self._tag_index[tag]:
                        self._tag_index[tag].remove(key)
                        if not self._tag_index[tag]:
                            del self._tag_index[tag]
                
                self._stats["invalidations"] += 1
                self._dirty = True
                
                lib_logger.debug(
                    f"ProviderCache[{self._cache_name}]: Invalidated key {key[:50]}..."
                )
                return True
            return False

    async def invalidate_pattern(self, pattern: str) -> int:
        """
        Invalidate all entries with keys matching a regex pattern.
        
        Args:
            pattern: Regex pattern to match keys against
            
        Returns:
            Number of entries invalidated
        """
        async with self._lock:
            compiled_pattern = re.compile(pattern)
            keys_to_remove = [
                key for key in self._cache.keys() if compiled_pattern.match(key)
            ]
            
            count = 0
            for key in keys_to_remove:
                entry = self._cache.pop(key)
                # Cleanup tag index
                for tag in entry.tags:
                    if tag in self._tag_index and key in self._tag_index[tag]:
                        self._tag_index[tag].remove(key)
                        if not self._tag_index[tag]:
                            del self._tag_index[tag]
                count += 1
            
            if count > 0:
                self._stats["pattern_invalidations"] += count
                self._dirty = True
                lib_logger.info(
                    f"ProviderCache[{self._cache_name}]: Invalidated {count} entries "
                    f"matching pattern {pattern}"
                )
            
            return count

    async def invalidate_prefix(self, prefix: str) -> int:
        """
        Invalidate all entries with keys starting with the given prefix.
        
        More efficient than regex for simple prefix matching.
        
        Returns:
            Number of entries invalidated
        """
        async with self._lock:
            keys_to_remove = [
                key for key in self._cache.keys() if key.startswith(prefix)
            ]
            
            count = 0
            for key in keys_to_remove:
                entry = self._cache.pop(key)
                # Cleanup tag index
                for tag in entry.tags:
                    if tag in self._tag_index and key in self._tag_index[tag]:
                        self._tag_index[tag].remove(key)
                        if not self._tag_index[tag]:
                            del self._tag_index[tag]
                count += 1
            
            if count > 0:
                self._stats["bulk_invalidations"] += count
                self._dirty = True
                lib_logger.info(
                    f"ProviderCache[{self._cache_name}]: Invalidated {count} entries "
                    f"with prefix {prefix[:50]}..."
                )
            
            return count

    async def invalidate_by_tag(self, tag: str) -> int:
        """
        Invalidate all entries associated with a specific tag.
        
        Args:
            tag: Tag to invalidate
            
        Returns:
            Number of entries invalidated
        """
        async with self._lock:
            if tag not in self._tag_index:
                return 0
            
            keys_to_remove = list(self._tag_index[tag])
            count = 0
            
            for key in keys_to_remove:
                if key in self._cache:
                    entry = self._cache.pop(key)
                    # Remove all tags for this entry from index
                    for entry_tag in entry.tags:
                        if entry_tag in self._tag_index and key in self._tag_index[entry_tag]:
                            self._tag_index[entry_tag].remove(key)
                            if not self._tag_index[entry_tag]:
                                del self._tag_index[entry_tag]
                    count += 1
            
            if count > 0:
                self._stats["tag_invalidations"] += count
                self._dirty = True
                lib_logger.info(
                    f"ProviderCache[{self._cache_name}]: Invalidated {count} entries "
                    f"with tag '{tag}'"
                )
            
            return count

    async def invalidate_older_than(self, timestamp: float) -> int:
        """
        Invalidate all entries created before the given timestamp.
        
        Args:
            timestamp: Unix timestamp threshold
            
        Returns:
            Number of entries invalidated
        """
        async with self._lock:
            keys_to_remove = [
                key for key, entry in self._cache.items() 
                if entry.timestamp < timestamp
            ]
            
            count = 0
            for key in keys_to_remove:
                entry = self._cache.pop(key)
                # Cleanup tag index
                for tag in entry.tags:
                    if tag in self._tag_index and key in self._tag_index[tag]:
                        self._tag_index[tag].remove(key)
                        if not self._tag_index[tag]:
                            del self._tag_index[tag]
                count += 1
            
            if count > 0:
                self._stats["bulk_invalidations"] += count
                self._dirty = True
                lib_logger.info(
                    f"ProviderCache[{self._cache_name}]: Invalidated {count} entries "
                    f"older than {timestamp}"
                )
            
            return count

    async def invalidate_all(self) -> int:
        """
        Clear all entries from the cache (memory and disk).
        
        Returns:
            Number of entries cleared
        """
        async with self._lock:
            count = len(self._cache)
            self._cache.clear()
            self._tag_index.clear()
            
            self._stats["bulk_invalidations"] += count
            self._dirty = True
            
            # Also clear disk if enabled
            if self._enable_disk and self._cache_file.exists():
                try:
                    async with self._disk_lock:
                        self._cache_file.unlink(missing_ok=True)
                except Exception as e:
                    lib_logger.error(
                        f"ProviderCache[{self._cache_name}]: Failed to clear disk cache: {e}"
                    )
            
            lib_logger.info(
                f"ProviderCache[{self._cache_name}]: Cleared all {count} entries"
            )
            return count

    async def touch(self, key: str, new_ttl: Optional[int] = None) -> bool:
        """
        Refresh the timestamp of a cache entry to extend its life.
        
        Args:
            key: Cache key to refresh
            new_ttl: Optional new TTL to set (not implemented in this version,
                     but resets the clock to now)
            
        Returns:
            True if entry existed and was refreshed, False otherwise
        """
        async with self._lock:
            if key in self._cache:
                entry = self._cache[key]
                entry.timestamp = time.time()
                entry.last_accessed = entry.timestamp
                self._dirty = True
                return True
            return False

    # =========================================================================
    # LRU EVICTION
    # =========================================================================

    async def _evict_lru_entries(self, count: int) -> None:
        """
        Evict the least recently used entries.
        
        This is called automatically when cache size exceeds max_memory_entries.
        """
        if count <= 0 or not self._cache:
            return
        
        # Sort by last_accessed (oldest first)
        sorted_entries = sorted(
            self._cache.items(),
            key=lambda x: x[1].last_accessed
        )
        
        evicted = 0
        for key, entry in sorted_entries[:count]:
            self._cache.pop(key, None)
            # Cleanup tag index
            for tag in entry.tags:
                if tag in self._tag_index and key in self._tag_index[tag]:
                    self._tag_index[tag].remove(key)
                    if not self._tag_index[tag]:
                        del self._tag_index[tag]
            evicted += 1
        
        self._stats["evictions"] += evicted
        if evicted > 0:
            lib_logger.debug(
                f"ProviderCache[{self._cache_name}]: LRU evicted {evicted} entries"
            )

    # =========================================================================
    # DISK PERSISTENCE
    # =========================================================================

    async def _persist_to_disk(self) -> None:
        """Persist current cache state to disk."""
        if not self._enable_disk or not self._dirty:
            return

        try:
            async with self._disk_lock:
                # Ensure directory exists
                self._cache_file.parent.mkdir(parents=True, exist_ok=True)
                
                # Build data structure
                now = time.time()
                entries = {}
                
                async with self._lock:
                    for key, entry in self._cache.items():
                        # Only persist entries that haven't exceeded disk TTL
                        if now - entry.timestamp <= self._disk_ttl:
                            entries[key] = entry.to_dict()
                
                data = {
                    "version": "2.0",
                    "timestamp": now,
                    "entry_count": len(entries),
                    "entries": entries,
                }
                
                await safe_write_json(self._cache_file, data)
                self._dirty = False
                self._stats["writes"] += 1
                self._disk_available = True
                
        except Exception as e:
            self._stats["disk_errors"] += 1
            self._disk_available = False
            lib_logger.error(
                f"ProviderCache[{self._cache_name}]: Disk write failed: {e}"
            )

    # =========================================================================
    # BACKGROUND TASKS
    # =========================================================================

    async def _start_background_tasks(self) -> None:
        """Start background writer and cleanup tasks."""
        if self._running:
            return
        
        self._running = True
        
        if self._enable_disk:
            self._writer_task = asyncio.create_task(
                self._writer_loop(),
                name=f"cache_writer_{self._cache_name}"
            )
        
        self._cleanup_task = asyncio.create_task(
            self._cleanup_loop(),
            name=f"cache_cleanup_{self._cache_name}"
        )

    async def _writer_loop(self) -> None:
        """Background task for periodic disk persistence."""
        while self._running:
            try:
                await asyncio.sleep(self._write_interval)
                if self._dirty:
                    await self._persist_to_disk()
            except asyncio.CancelledError:
                break
            except Exception as e:
                lib_logger.error(
                    f"ProviderCache[{self._cache_name}]: Writer error: {e}"
                )

    async def _cleanup_loop(self) -> None:
        """Background task for periodic cleanup of expired entries."""
        while self._running:
            try:
                await asyncio.sleep(self._cleanup_interval)
                await self._cleanup_expired()
            except asyncio.CancelledError:
                break
            except Exception as e:
                lib_logger.error(
                    f"ProviderCache[{self._cache_name}]: Cleanup error: {e}"
                )

    async def _cleanup_expired(self) -> int:
        """
        Remove memory-expired entries (those exceeding memory_ttl).
        Note: Disk entries are preserved until disk_ttl.
        
        Returns:
            Number of entries cleaned up
        """
        async with self._lock:
            now = time.time()
            expired_keys = [
                key for key, entry in self._cache.items()
                if now - entry.timestamp > self._memory_ttl
            ]
            
            for key in expired_keys:
                entry = self._cache.pop(key)
                # Cleanup tag index
                for tag in entry.tags:
                    if tag in self._tag_index and key in self._tag_index[tag]:
                        self._tag_index[tag].remove(key)
                        if not self._tag_index[tag]:
                            del self._tag_index[tag]
            
            if expired_keys:
                lib_logger.debug(
                    f"ProviderCache[{self._cache_name}]: Cleaned up {len(expired_keys)} "
                    f"expired entries"
                )
            
            return len(expired_keys)

    # =========================================================================
    # STATISTICS & MONITORING
    # =========================================================================

    def get_stats(self) -> Dict[str, Any]:
        """
        Get current cache statistics.
        
        Returns:
            Dictionary with cache statistics
        """
        total_hits = self._stats["memory_hits"] + self._stats["disk_hits"]
        total_requests = total_hits + self._stats["misses"]
        hit_rate = total_hits / total_requests if total_requests > 0 else 0
        
        return {
            "cache_name": self._cache_name,
            "memory_entries": len(self._cache),
            "tag_count": len(self._tag_index),
            "hit_rate": round(hit_rate, 4),
            "total_requests": total_requests,
            **self._stats,
            "disk_enabled": self._enable_disk,
            "disk_available": self._disk_available,
            "max_memory_entries": self._max_memory_entries,
            "memory_ttl": self._memory_ttl,
            "disk_ttl": self._disk_ttl,
        }

    def get_cache_info(self) -> Dict[str, Any]:
        """
        Get detailed information about cache contents (for debugging).
        
        Returns:
            Dictionary with cache metadata
        """
        now = time.time()
        
        # Calculate age distribution
        ages = [now - entry.timestamp for entry in self._cache.values()]
        access_counts = [entry.access_count for entry in self._cache.values()]
        
        info = {
            "cache_name": self._cache_name,
            "total_entries": len(self._cache),
            "unique_tags": len(self._tag_index),
            "avg_age_seconds": sum(ages) / len(ages) if ages else 0,
            "max_age_seconds": max(ages) if ages else 0,
            "avg_access_count": sum(access_counts) / len(access_counts) if access_counts else 0,
            "memory_usage_estimate_bytes": sum(
                len(k) + len(e.value) for k, e in self._cache.items()
            ),
        }
        
        return info

    async def shutdown(self) -> None:
        """Gracefully shutdown the cache, persisting any pending data."""
        lib_logger.info(f"ProviderCache[{self._cache_name}]: Shutting down...")
        self._running = False
        
        # Cancel background tasks
        if self._writer_task:
            self._writer_task.cancel()
            try:
                await self._writer_task
            except asyncio.CancelledError:
                pass
        
        if self._cleanup_task:
            self._cleanup_task.cancel()
            try:
                await self._cleanup_task
            except asyncio.CancelledError:
                pass
        
        # Final persist
        if self._enable_disk and self._dirty:
            await self._persist_to_disk()
        
        lib_logger.info(f"ProviderCache[{self._cache_name}]: Shutdown complete")

    # =========================================================================
    # CONTEXT MANAGER
    # =========================================================================

    async def __aenter__(self) -> "ProviderCache":
        """Async context manager entry."""
        return self

    async def __aexit__(self, exc_type, exc_val, exc_tb) -> None:
        """Async context manager exit."""
        await self.shutdown()
