"""Dynamic Model Scoring Engine

Calculates real-time scores for models based on agentic coding performance,
tokens per second, and availability with configurable weights and thresholds.
"""

import time
import sqlite3
from typing import Dict, List, Optional, Tuple
from dataclasses import dataclass, field
from pathlib import Path
from enum import Enum
import logging
import yaml

logger = logging.getLogger(__name__)


class ModelCategory(Enum):
    """Model categories for scoring."""
    CODING_ELITE = "coding-elite"
    CODING_SMART = "coding-smart"
    CODING_FAST = "coding-fast"
    CHAT_ELITE = "chat-elite"
    CHAT_SMART = "chat-smart"
    CHAT_FAST = "chat-fast"
    ROLEPLAY = "chat-rp"
    SPECIFIC = "specific"  # User-specified models


class SkipCondition(Enum):
    """Conditions that cause a model to be excluded from ranking."""
    RATE_LIMITED = "rate_limited"
    USAGE_LIMIT = "usage_limit"
    UNHEALTHY = "unhealthy"
    DOWN = "down"


@dataclass
class ModelScore:
    """Complete score breakdown for a model."""
    provider: str
    model: str
    category: str
    # Component scores (0-100 scale)
    agentic_score: float = 0.0
    intelligence_score: float = 0.0
    tps_score: float = 0.0
    hallucination_rate: float = 10.0
    hallucination_penalty: float = 0.0
    # Availability (yes/no filter)
    is_available: bool = True
    web_search_capable: bool = False
    # Skip conditions
    skip_reason: Optional[str] = None
    # Weighted total
    total_score: float = 0.0
    meets_threshold: bool = True


@dataclass
class ScoringConfig:
    """Scoring configuration loaded from config file."""
    virtual_models: Dict[str, float] = field(default_factory=lambda: {
        "agentic_score": 0.70,
        "tps_score": 0.30,
        "availability_score": 0.0,
    })
    specific_models: Dict[str, float] = field(default_factory=lambda: {
        "tps_score": 0.70,
        "agentic_score": 0.30,
        "availability_score": 0.0,
    })
    roleplay: Dict[str, float] = field(default_factory=lambda: {
        "tps_score": 0.80,
        "agentic_score": 0.20,
        "availability_score": 0.0,
    })
    # Skip condition thresholds
    consecutive_failures_threshold: int = 5
    failure_rate_threshold: float = 0.50
    # TPS normalization
    max_observed_tps: float = 3000.0
    # Hallucination penalty scale (higher rate = more penalty)
    hallucination_penalty_scale: float = 10.0


class DynamicScoringEngine:
    """Calculates dynamic scores for virtual model fallback ordering.

    Models are ordered purely by performance benchmarks + TPS + hallucination.
    Rate limits and availability are NOT used for ordering — rate-limited
    providers are skipped at runtime instead.
    """

    # Default weights for different model categories (from config)
    DEFAULT_WEIGHTS = {
        ModelCategory.CODING_ELITE: {
            "agentic": 0.75,
            "tps": 0.15,
            "hallucination_penalty": 0.10,
        },
        ModelCategory.CODING_SMART: {
            "agentic": 0.65,
            "tps": 0.20,
            "hallucination_penalty": 0.15,
        },
        ModelCategory.CODING_FAST: {
            "agentic": 0.30,
            "tps": 0.55,
            "hallucination_penalty": 0.15,
        },
        ModelCategory.CHAT_ELITE: {
            "intelligence": 0.80,
            "hallucination_penalty": 0.20,
        },
        ModelCategory.CHAT_SMART: {
            "intelligence": 0.50,
            "tps": 0.30,
            "hallucination_penalty": 0.20,
        },
        ModelCategory.CHAT_FAST: {
            "tps": 0.60,
            "intelligence": 0.20,
            "hallucination_penalty": 0.20,
        },
        ModelCategory.ROLEPLAY: {
            "tps": 0.80,
            "agentic": 0.20,
            "hallucination_penalty": 0.0,
        },
        ModelCategory.SPECIFIC: {
            "tps": 0.70,
            "agentic": 0.30,
            "hallucination_penalty": 0.0,
        },
    }

    # Thresholds for minimum scores (models below threshold are flagged)
    THRESHOLDS = {
        ModelCategory.CODING_ELITE: 75.0,
        ModelCategory.CODING_SMART: 65.0,
        ModelCategory.CODING_FAST: 40.0,
        ModelCategory.CHAT_ELITE: 0.0,
        ModelCategory.CHAT_SMART: 0.0,
        ModelCategory.CHAT_FAST: 0.0,
        ModelCategory.ROLEPLAY: 0.0,
        ModelCategory.SPECIFIC: 0.0,
    }

    def __init__(
        self,
        telemetry_manager=None,
        model_rankings_path: str = "config/model_rankings.yaml",
        scoring_config_path: str = "config/scoring_config.yaml",
    ):
        self.telemetry = telemetry_manager
        self.model_rankings_path = Path(model_rankings_path)
        self.scoring_config_path = Path(scoring_config_path)
        self._rankings_cache: Dict[str, Dict] = {}
        self._scoring_config: Optional[ScoringConfig] = None
        self._last_refresh = 0
        self._config_last_refresh = 0

    def load_scoring_config(self) -> ScoringConfig:
        """Load scoring configuration from YAML file."""
        try:
            with open(self.scoring_config_path) as f:
                data = yaml.safe_load(f)

            scoring = data.get("scoring", {})
            
            # Build config from YAML
            config = ScoringConfig()
            
            # Virtual models weights
            vm_weights = scoring.get("virtual_models", {}).get("weights", {})
            config.virtual_models = {
                "agentic_score": vm_weights.get("agentic_score", 0.70),
                "tps_score": vm_weights.get("tps_score", 0.30),
                "availability_score": vm_weights.get("availability_score", 0.0),
            }
            
            # Specific models weights
            sm_weights = scoring.get("specific_models", {}).get("weights", {})
            config.specific_models = {
                "tps_score": sm_weights.get("tps_score", 0.70),
                "agentic_score": sm_weights.get("agentic_score", 0.30),
                "availability_score": sm_weights.get("availability_score", 0.0),
            }
            
            # Roleplay weights
            rp_weights = scoring.get("roleplay", {}).get("weights", {})
            config.roleplay = {
                "tps_score": rp_weights.get("tps_score", 0.80),
                "agentic_score": rp_weights.get("agentic_score", 0.20),
                "availability_score": rp_weights.get("availability_score", 0.0),
            }
            
            # Availability settings
            avail = scoring.get("availability", {})
            config.consecutive_failures_threshold = avail.get("consecutive_failures_threshold", 5)
            config.failure_rate_threshold = avail.get("failure_rate_threshold", 0.50)
            
            # TPS settings
            tps_config = scoring.get("tps", {})
            config.max_observed_tps = tps_config.get("max_observed_tps", 3000.0)
            
            self._scoring_config = config
            self._config_last_refresh = time.time()
            return config
            
        except Exception as e:
            logger.error(f"Failed to load scoring config: {e}")
            # Return defaults
            return ScoringConfig()

    def get_scoring_config(self) -> ScoringConfig:
        """Get scoring config, loading if necessary."""
        if self._scoring_config is None or time.time() - self._config_last_refresh > 3600:
            return self.load_scoring_config()
        return self._scoring_config

    def load_model_rankings(self) -> Dict[str, Dict]:
        """Load model rankings from YAML file."""
        try:
            with open(self.model_rankings_path) as f:
                data = yaml.safe_load(f)

            rankings = {}
            for model in data.get("models", []):
                model_id = model.get("id", "")
                if "/" in model_id:
                    provider, model_name = model_id.split("/", 1)
                    key = f"{provider}/{model_name}"
                    rankings[key] = model

            self._rankings_cache = rankings
            self._last_refresh = time.time()
            return rankings

        except Exception as e:
            logger.error(f"Failed to load model rankings: {e}")
            return {}

    def _ensure_rankings_loaded(self):
        """Ensure rankings are loaded, refreshing if stale."""
        if not self._rankings_cache or time.time() - self._last_refresh > 3600:
            self.load_model_rankings()

    def get_model_score(self, provider: str, model: str) -> Optional[float]:
        """Get agentic coding score for a model."""
        self._ensure_rankings_loaded()

        key = f"{provider}/{model}"
        model_data = self._rankings_cache.get(key)

        if not model_data:
            model_data = self._find_model_by_name_or_id(model)

        if model_data:
            scores = model_data.get("scores", {})
            return scores.get("agentic_coding", scores.get("swe_bench", 0.0))

        return None

    def _find_model_by_name_or_id(self, model: str) -> Optional[Dict]:
        """Find model data by name or ID suffix."""
        for k, v in self._rankings_cache.items():
            if (
                k.endswith(f"/{model}")
                or v.get("name", "").lower() == model.lower()
            ):
                return v
        return None

    def get_intelligence_score(self, provider: str, model: str) -> float:
        """Get general intelligence score for a model (for chat models)."""
        self._ensure_rankings_loaded()

        key = f"{provider}/{model}"
        model_data = self._rankings_cache.get(key)

        if not model_data:
            model_data = self._find_model_by_name_or_id(model)

        if model_data:
            scores = model_data.get("scores", {})
            # Try various intelligence metrics
            return scores.get("intelligence", 
                   scores.get("mmlu", 
                   scores.get("agentic_coding", 0.0)))

        return 0.0

    def get_hallucination_rate(self, provider: str, model: str) -> float:
        """Get hallucination rate for a model (lower is better, 0-100 scale)."""
        self._ensure_rankings_loaded()

        key = f"{provider}/{model}"
        model_data = self._rankings_cache.get(key)

        if not model_data:
            model_data = self._find_model_by_name_or_id(model)

        if model_data:
            scores = model_data.get("scores", {})
            return scores.get("hallucination_rate", 10.0)

        return 10.0

    def get_web_search_capable(self, provider: str, model: str) -> bool:
        """Check if model supports web search capability."""
        self._ensure_rankings_loaded()

        key = f"{provider}/{model}"
        model_data = self._rankings_cache.get(key)

        if not model_data:
            model_data = self._find_model_by_name_or_id(model)

        if model_data:
            capabilities = model_data.get("capabilities", {})
            return capabilities.get("web_search_capable", False)

        return False

    def get_tps_estimate(self, provider: str, model: str) -> float:
        """Get TPS from telemetry or return default estimate."""
        if not self.telemetry:
            # Return default estimate based on known fast models
            return self._get_default_tps(model)

        try:
            conn = sqlite3.connect(self.telemetry.db_path)
            cursor = conn.cursor()

            # Query average TPS from recent successful requests
            cursor.execute(
                """
                SELECT AVG(tps) as avg_tps
                FROM request_logs
                WHERE provider = ? AND model = ? 
                AND timestamp > datetime('now', '-1 hour')
                AND status = 'success'
                """,
                (provider, model),
            )

            row = cursor.fetchone()
            conn.close()

            if row and row[0]:
                return float(row[0])

        except Exception as e:
            logger.debug(f"Failed to get TPS from telemetry: {e}")

        return self._get_default_tps(model)

    def _get_default_tps(self, model: str) -> float:
        """Get default TPS estimate based on known model speeds."""
        # Known fast models
        fast_models = {
            "claude-3-5-sonnet": 150.0,
            "claude-3-opus": 80.0,
            "gpt-4o": 120.0,
            "gpt-4o-mini": 200.0,
            "gemini-1.5-pro": 100.0,
            "gemini-1.5-flash": 250.0,
            "llama-3.1-405b": 80.0,
            "llama-3.1-70b": 120.0,
            "llama-3.1-8b": 180.0,
            "mixtral-8x7b": 150.0,
        }
        
        model_lower = model.lower()
        for known, tps in fast_models.items():
            if known in model_lower:
                return tps
        
        return 100.0  # Default fallback

    def normalize_tps(self, tps: float, max_tps: Optional[float] = None) -> float:
        """Normalize TPS to 0-100 scale."""
        if max_tps is None:
            config = self.get_scoring_config()
            max_tps = config.max_observed_tps
        
        return min(100.0, (tps / max_tps) * 100.0)

    def calculate_hallucination_penalty(self, hallucination_rate: float) -> float:
        """Calculate penalty based on hallucination rate (0-100).
        
        Higher hallucination rate = higher penalty.
        Penalty is scaled so that 0% hallucination = 0 penalty,
        100% hallucination = 100 penalty.
        """
        config = self.get_scoring_config()
        scale = config.hallucination_penalty_scale
        return min(100.0, hallucination_rate * (100.0 / scale))

    def get_model_category(self, model: str, is_virtual: bool = True) -> ModelCategory:
        """Determine the category for a model based on its name."""
        model_lower = model.lower()
        
        # Roleplay models
        if "rp" in model_lower or "roleplay" in model_lower:
            return ModelCategory.ROLEPLAY
        
        # User-specified (specific) models
        if not is_virtual:
            return ModelCategory.SPECIFIC
        
        # Virtual model categories based on name suffixes
        if "-elite" in model_lower:
            if "code" in model_lower or "coding" in model_lower:
                return ModelCategory.CODING_ELITE
            return ModelCategory.CHAT_ELITE
        
        if "-smart" in model_lower:
            if "code" in model_lower or "coding" in model_lower:
                return ModelCategory.CODING_SMART
            return ModelCategory.CHAT_SMART
        
        if "-fast" in model_lower:
            if "code" in model_lower or "coding" in model_lower:
                return ModelCategory.CODING_FAST
            return ModelCategory.CHAT_FAST
        
        # Default to chat-smart for unknown virtual models
        return ModelCategory.CHAT_SMART

    def get_weights_for_category(self, category: ModelCategory) -> Dict[str, float]:
        """Get scoring weights for a model category."""
        # First try to get from config
        config = self.get_scoring_config()
        
        if category == ModelCategory.ROLEPLAY:
            return {
                "tps": config.roleplay.get("tps_score", 0.80),
                "agentic": config.roleplay.get("agentic_score", 0.20),
                "intelligence": 0.0,
                "hallucination_penalty": 0.0,
            }
        elif category == ModelCategory.SPECIFIC:
            return {
                "tps": config.specific_models.get("tps_score", 0.70),
                "agentic": config.specific_models.get("agentic_score", 0.30),
                "intelligence": 0.0,
                "hallucination_penalty": 0.0,
            }
        elif category in (ModelCategory.CODING_ELITE, ModelCategory.CODING_SMART, 
                         ModelCategory.CODING_FAST):
            return {
                "agentic": config.virtual_models.get("agentic_score", 0.70),
                "tps": config.virtual_models.get("tps_score", 0.30),
                "intelligence": 0.0,
                "hallucination_penalty": 0.10,
            }
        else:  # Chat categories
            return {
                "agentic": config.virtual_models.get("agentic_score", 0.70),
                "tps": config.virtual_models.get("tps_score", 0.30),
                "intelligence": 0.0,
                "hallucination_penalty": 0.10,
            }

    def calculate_model_score(
        self,
        provider: str,
        model: str,
        is_virtual: bool = True,
        is_available: bool = True,
        skip_reason: Optional[str] = None,
    ) -> ModelScore:
        """Calculate complete score for a model.
        
        Args:
            provider: Provider name
            model: Model name
            is_virtual: Whether this is a virtual model (determines category)
            is_available: Whether the model is available (yes/no filter)
            skip_reason: If set, model should be skipped
            
        Returns:
            ModelScore dataclass with all components and total
        """
        category = self.get_model_category(model, is_virtual)
        weights = self.get_weights_for_category(category)
        
        # Get component scores
        agentic_score = self.get_model_score(provider, model) or 0.0
        intelligence_score = self.get_intelligence_score(provider, model) or 0.0
        hallucination_rate = self.get_hallucination_rate(provider, model)
        hallucination_penalty = self.calculate_hallucination_penalty(hallucination_rate)
        
        raw_tps = self.get_tps_estimate(provider, model)
        tps_score = self.normalize_tps(raw_tps)
        
        web_search_capable = self.get_web_search_capable(provider, model)
        
        # Check if model should be skipped
        should_skip = skip_reason is not None or not is_available
        if should_skip and skip_reason is None:
            skip_reason = "unavailable" if not is_available else "unknown"
        
        # Get threshold for category
        threshold = self.THRESHHOLDS.get(category, 0.0)
        
        # Calculate weighted total score (only if not skipped)
        total_score = 0.0
        if not should_skip:
            # Use agentic for coding, intelligence for chat
            if category in (ModelCategory.CODING_ELITE, ModelCategory.CODING_SMART, 
                           ModelCategory.CODING_FAST):
                primary_score = agentic_score
            else:
                primary_score = intelligence_score if intelligence_score > 0 else agentic_score
            
            # Apply weights
            total_score = (
                weights.get("agentic", 0.0) * agentic_score +
                weights.get("intelligence", 0.0) * intelligence_score +
                weights.get("tps", 0.0) * tps_score -
                weights.get("hallucination_penalty", 0.0) * hallucination_penalty
            )
        
        meets_threshold = total_score >= threshold if not should_skip else False
        
        return ModelScore(
            provider=provider,
            model=model,
            category=category.value,
            agentic_score=agentic_score,
            intelligence_score=intelligence_score,
            tps_score=tps_score,
            hallucination_rate=hallucination_rate,
            hallucination_penalty=hallucination_penalty,
            is_available=is_available,
            web_search_capable=web_search_capable,
            skip_reason=skip_reason,
            total_score=total_score,
            meets_threshold=meets_threshold,
        )

    def rank_models(
        self,
        models: List[Tuple[str, str]],  # List of (provider, model)
        is_virtual: bool = True,
        availability_status: Optional[Dict[str, bool]] = None,
        skip_reasons: Optional[Dict[str, Optional[str]]] = None,
    ) -> List[ModelScore]:
        """Rank a list of models by their total score.
        
        Args:
            models: List of (provider, model) tuples
            is_virtual: Whether these are virtual models
            availability_status: Dict mapping "provider/model" to availability
            skip_reasons: Dict mapping "provider/model" to skip reason
            
        Returns:
            List of ModelScore sorted by total_score descending
        """
        availability_status = availability_status or {}
        skip_reasons = skip_reasons or {}
        
        scores = []
        for provider, model in models:
            key = f"{provider}/{model}"
            
            is_available = availability_status.get(key, True)
            skip_reason = skip_reasons.get(key)
            
            score = self.calculate_model_score(
                provider=provider,
                model=model,
                is_virtual=is_virtual,
                is_available=is_available,
                skip_reason=skip_reason,
            )
            scores.append(score)
        
        # Sort by total score descending, but keep skipped models at the end
        scores.sort(key=lambda s: (s.skip_reason is not None, -s.total_score))
        
        return scores

    def should_skip_model(
        self,
        provider: str,
        model: str,
        consecutive_failures: int = 0,
        failure_rate: float = 0.0,
        is_rate_limited: bool = False,
        is_quota_exhausted: bool = False,
    ) -> Optional[str]:
        """Determine if a model should be skipped based on conditions.
        
        Args:
            provider: Provider name
            model: Model name  
            consecutive_failures: Number of consecutive failures
            failure_rate: Ratio of failures to total requests (0.0-1.0)
            is_rate_limited: Whether provider is currently rate limited
            is_quota_exhausted: Whether quota is exhausted
            
        Returns:
            Skip reason string if should skip, None otherwise
        """
        config = self.get_scoring_config()
        
        # Check rate limit (skip condition from config)
        if is_rate_limited:
            return SkipCondition.RATE_LIMITED.value
        
        # Check quota exhaustion
        if is_quota_exhausted:
            return SkipCondition.USAGE_LIMIT.value
        
        # Check consecutive failures threshold
        if consecutive_failures >= config.consecutive_failures_threshold:
            return SkipCondition.UNHEALTHY.value
        
        # Check failure rate threshold
        if failure_rate >= config.failure_rate_threshold:
            return SkipCondition.DOWN.value
        
        return None
