# --- Security Headers Middleware ---
@app.middleware("http")
async def add_security_headers(request: Request, call_next):
    """Add security headers to all responses as per Phase 1.3 Security Hardening."""
    response = await call_next(request)
    
    # Prevent MIME type sniffing
    response.headers["X-Content-Type-Options"] = "nosniff"
    
    # Prevent clickjacking attacks
    response.headers["X-Frame-Options"] = "DENY"
    
    # XSS Protection for legacy browsers
    response.headers["X-SS-Protection"] = "1; mode=block"
    
    # Enforce HTTPS connections (1 year max-age, include subdomains)
    response.headers["Strict-Transport-Security"] = "max-age=31536000; includeSubDomains"
    
    # Control referrer information leakage
    response.headers["Referrer-Policy"] = "strict-origin-when-cross-origin"
    
    # Restrict Flash/PDF cross-domain policies
    response.headers["X-Permitted-Cross-Domain-Policies"] = "none"
    
    # Feature policy / Permissions policy to restrict browser features
    response.headers["Permissions-Policy"] = (
        "accelerometer=(), camera=(), geolocation=(), gyroscope=(), "
        "magnetometer=(), microphone=(), payment=(), usb=(), "
        "interest-cohort=(), browsing-topics=()"
    )
    
    # Content Security Policy for API (restrictive default)
    response.headers["Content-Security-Policy"] = (
        "default-src 'none'; frame-ancestors 'none'; base-uri 'none'; "
        "form-action 'none'"
    )
    
    # Disable caching for sensitive API responses by default
    # (Can be overridden by specific endpoints if needed)
    response.headers["Cache-Control"] = "no-store, no-cache, must-revalidate, proxy-revalidate"
    response.headers["Pragma"] = "no-cache"
    response.headers["Expires"] = "0"
    
    return response

# --- API Key Security ---
api_key_header = APIKeyHeader(name="Authorization", auto_error=False)

async def verify_api_key(authorization: Optional[str] = Depends(api_key_header)):
    """Verify the API key from Authorization header."""
    if not proxy_api_key:
        return True  # No key required if not set
    
    if not authorization:
        raise HTTPException(status_code=401, detail="Missing Authorization header")
    
    # Handle Bearer token format
    if authorization.startswith("Bearer "):
        token = authorization.replace("Bearer ", "")
    else:
        token = authorization
    
    if token != proxy_api_key:
        raise HTTPException(status_code=403, detail="Invalid API key")
    
    return True
