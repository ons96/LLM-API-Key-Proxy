import asyncio
import time
from typing import List, Dict, Any, Optional, Union
import logging

logger = logging.getLogger(__name__)


class EmbeddingBatcher:
    """
    Batches and processes embedding requests efficiently.
    For Phase 4.3, implements immediate processing with batching architecture
    ready for future optimization.
    """
    
    def __init__(self, rotator_client, max_batch_size: int = 100, max_wait_time: float = 0.01):
        self.rotator_client = rotator_client
        self.max_batch_size = max_batch_size
        self.max_wait_time = max_wait_time
        self._queue = []
        self._lock = asyncio.Lock()
        self._processing = False
        self._batch_task = None
        
    async def start(self):
        """Start the background batch processing."""
        self._processing = True
        
    async def stop(self):
        """Stop processing and flush remaining items."""
        self._processing = False
        if self._batch_task and not self._batch_task.done():
            self._batch_task.cancel()
            try:
                await self._batch_task
            except asyncio.CancelledError:
                pass
        await self.flush()
        
    async def submit(self, model: str, input_data: Union[str, List[str]], **kwargs) -> Dict[str, Any]:
        """
        Submit an embedding request.
        
        Args:
            model: The model to use for embeddings
            input_data: Single string or list of strings to embed
            **kwargs: Additional arguments (dimensions, user, etc.)
            
        Returns:
            OpenAI-compatible embedding response dict
        """
        # Normalize input to list for consistent processing
        if isinstance(input_data, str):
            input_list = [input_data]
        else:
            input_list = input_data
            
        try:
            # Direct processing through rotator client
            # Future: Implement actual batching logic here
            response = await self.rotator_client.aembedding(
                model=model,
                input=input_list,
                **kwargs
            )
            
            # Ensure OpenAI-compatible format
            if isinstance(response, dict) and "data" in response:
                return response
            
            # Handle non-standard response format
            if isinstance(response, dict):
                return {
                    "object": "list",
                    "data": [
                        {
                            "object": "embedding",
                            "embedding": response.get("embedding", response.get("embeddings", [])),
                            "index": i
                        }
                        for i in range(len(input_list))
                    ],
                    "model": model,
                    "usage": response.get("usage", {
                        "prompt_tokens": sum(len(str(x)) for x in input_list),
                        "total_tokens": sum(len(str(x)) for x in input_list)
                    })
                }
            
            return {
                "object": "list",
                "data": [],
                "model": model,
                "usage": {"prompt_tokens": 0, "total_tokens": 0}
            }
                
        except Exception as e:
            logger.error(f"Embedding request failed: {e}")
            raise
