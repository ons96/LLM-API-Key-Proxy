import httpx
import logging
from typing import List, Dict, Any, Optional
from .provider_interface import ProviderInterface

lib_logger = logging.getLogger('rotator_library')
lib_logger.propagate = False
if not lib_logger.handlers:
    lib_logger.addHandler(logging.NullHandler())

class CohereProvider(ProviderInterface):
    """
    Provider implementation for the Cohere API.
    
    Free Tier Details:
    - 1,000 API calls/month
    - 20 calls/minute rate limit
    - Access to command-light and embed models
    - No credit card required for signup
    """
    
    # Known free tier models as of Phase 5.1 research
    FREE_TIER_MODELS = {
        "command-light",
        "command-light-nightly",
        "embed-english-v2.0",
        "embed-english-light-v2.0",
        "embed-multilingual-v2.0"
    }
    
    async def get_models(self, api_key: str, client: httpx.AsyncClient) -> List[str]:
        """
        Fetches the list of available models from the Cohere API.
        Handles free tier specific rate limiting and authentication errors.
        """
        try:
            response = await client.get(
                "https://api.cohere.ai/v1/models",
                headers={"Authorization": f"Bearer {api_key}"},
                timeout=30.0
            )
            
            # Handle free tier rate limiting (429)
            if response.status_code == 429:
                retry_after = response.headers.get("retry-after", "unknown")
                lib_logger.warning(f"Cohere free tier rate limit hit. Retry after: {retry_after}s")
                return []
            
            response.raise_for_status()
            data = response.json()
            
            models = []
            for model in data.get("models", []):
                model_name = model.get("name", "")
                if model_name:
                    models.append(f"cohere/{model_name}")
            
            # Log rate limit status for free tier monitoring
            remaining = response.headers.get("x-ratelimit-remaining")
            if remaining:
                lib_logger.debug(f"Cohere API calls remaining: {remaining}")
            
            return models
            
        except httpx.HTTPStatusError as e:
            if e.response.status_code == 401:
                lib_logger.error("Cohere API key invalid or expired")
            elif e.response.status_code == 429:
                lib_logger.error("Cohere free tier rate limit exceeded (20/min or 1000/month)")
            else:
                lib_logger.error(f"Cohere API error {e.response.status_code}: {e}")
            return []
        except httpx.TimeoutException:
            lib_logger.error("Cohere API request timed out")
            return []
        except httpx.RequestError as e:
            lib_logger.error(f"Network error connecting to Cohere: {e}")
            return []
        except Exception as e:
            lib_logger.error(f"Unexpected error fetching Cohere models: {e}")
            return []
    
    async def validate_key(self, api_key: str, client: httpx.AsyncClient) -> Dict[str, Any]:
        """
        Validates if the API key is active and determines if it's free tier.
        Returns validation status and tier information.
        """
        try:
            response = await client.get(
                "https://api.cohere.ai/v1/models",
                headers={"Authorization": f"Bearer {api_key}"},
                timeout=10.0
            )
            
            if response.status_code == 200:
                # Check headers to infer tier
                limit = response.headers.get("x-ratelimit-limit", "")
                remaining = response.headers.get("x-ratelimit-remaining", "")
                
                # Free tier typically has 20/min or 1000/month limits
                is_likely_free = (
                    "20" in limit or 
                    "1000" in limit or 
                    (remaining.isdigit() and int(remaining) <= 1000)
                )
                
                return {
                    "valid": True,
                    "is_free_tier": is_likely_free,
                    "rate_limit": limit,
                    "remaining": remaining
                }
            elif response.status_code == 401:
                return {"valid": False, "error": "Invalid API key"}
            elif response.status_code == 429:
                return {"valid": True, "error": "Rate limited", "is_free_tier": True}
            else:
                return {"valid": False, "error": f"HTTP {response.status_code}"}
                
        except Exception as e:
            return {"valid": False, "error": str(e)}
    
    def filter_free_tier_models(self, models: List[str]) -> List[str]:
        """
        Filter model list to only include those available on free tier.
        """
        return [
            model for model in models 
            if any(free_model in model for free_model in self.FREE_TIER_MODELS)
        ]
