      matrix:
        python-version: ["3.9", "3.10", "3.11", "3.12"]

    steps:
    - uses: actions/checkout@v4
    
    - name: Set up Python ${{ matrix.python-version }}
      uses: actions/setup-python@v5
      with:
        python-version: ${{ matrix.python-version }}
        
    - name: Install dependencies
      run: |
        python -m pip install --upgrade pip
        pip install pytest pytest-asyncio httpx
        # Install production dependencies (adjust if requirements.txt exists)
        pip install fastapi uvicorn litellm pydantic python-dotenv colorlog rich
        # Install the package in editable mode
        pip install -e .
        
    - name: Run tests with pytest
      run: |
        pytest src/ -v --tb=short
        
    - name: Check code formatting with ruff
      run: |
        pip install ruff
        ruff check src/ --select E,W,F || true
        
    - name: Type checking with mypy (optional)
      run: |
        pip install mypy
        mypy src/ --ignore-missing-imports || true
