import logging
import os
from typing import Dict, Optional, Any, List
from pathlib import Path
import yaml

logger = logging.getLogger("proxy_app.router_core")


class VirtualModelResolver:
    """
    Resolves virtual model names to concrete provider model IDs.
    Loads mappings from config/virtual_models.yaml and related files.
    """

    def __init__(self, config_path: Optional[str] = None):
        self.config_path = config_path or self._default_config_path()
        self.virtual_models: Dict[str, Any] = {}
        self.resolution_cache: Dict[str, str] = {}
        self._load_virtual_models()

    def _default_config_path(self) -> Path:
        """Determine default config path based on project structure."""
        current_file = Path(__file__).resolve()
        project_root = current_file.parent.parent.parent
        return project_root / "config" / "virtual_models.yaml"

    def _load_virtual_models(self):
        """Load virtual model mappings from configuration files."""
        logger.info(f"Loading virtual model configuration from {self.config_path}")

        if not self.config_path.exists():
            logger.warning(f"Virtual models config not found at {self.config_path}")
            return

        try:
            with open(self.config_path, "r") as f:
                config = yaml.safe_load(f)

            if not config:
                logger.warning("Virtual models config is empty")
                return

            self.virtual_models = config.get("virtual_models", {})
            aliases = config.get("aliases", {})

            logger.info(
                f"Loaded {len(self.virtual_models)} virtual models and "
                f"{len(aliases)} aliases from configuration"
            )

            # Log each virtual model at debug level for troubleshooting
            for vname, mapping in self.virtual_models.items():
                targets = (
                    mapping.get("targets", []) if isinstance(mapping, dict) else mapping
                )
                logger.debug(f"Registered virtual model '{vname}' -> {targets}")

            # Load backup config if primary is empty or for fallback
            self._load_backup_config()

        except yaml.YAMLError as e:
            logger.error(f"Failed to parse virtual models YAML: {e}", exc_info=True)
        except Exception as e:
            logger.error(f"Unexpected error loading virtual models: {e}", exc_info=True)

    def _load_backup_config(self):
        """Load backup configuration for redundancy."""
        backup_path = self.config_path.parent / "virtual_models_backup.yaml"
        if backup_path.exists():
            try:
                with open(backup_path, "r") as f:
                    backup = yaml.safe_load(f)
                if backup:
                    logger.debug(f"Loaded backup virtual models from {backup_path}")
            except Exception as e:
                logger.warning(f"Failed to load backup config: {e}")

    def resolve(
        self, model_name: str, preferred_provider: Optional[str] = None
    ) -> Optional[str]:
        """
        Resolve a virtual model name to a concrete model ID.

        Args:
            model_name: The requested model name (virtual or concrete)
            preferred_provider: Optional preferred provider hint

        Returns:
            Resolved concrete model ID or original name if not a virtual model
        """
        logger.debug(
            f"Attempting to resolve model: {model_name} (preferred_provider: {preferred_provider})"
        )

        # Check cache first
        cache_key = f"{model_name}:{preferred_provider}"
        if cache_key in self.resolution_cache:
            resolved = self.resolution_cache[cache_key]
            logger.debug(f"Cache hit for '{model_name}' -> '{resolved}'")
            return resolved

        # Check if this is a virtual model
        if model_name not in self.virtual_models:
            logger.debug(f"'{model_name}' is not a virtual model, returning as-is")
            return model_name

        mapping = self.virtual_models[model_name]
        logger.info(
            f"Resolving virtual model '{model_name}' with mapping type: {type(mapping).__name__}"
        )

        try:
            if isinstance(mapping, str):
                # Simple string mapping
                resolved = mapping
                logger.info(
                    f"Virtual model '{model_name}' resolved to '{resolved}' (direct mapping)"
                )

            elif isinstance(mapping, dict):
                # Complex mapping with targets, weights, etc.
                targets = mapping.get("targets", [])
                if not targets:
                    logger.error(
                        f"Virtual model '{model_name}' has empty targets list"
                    )
                    return None

                # Apply provider preference if specified
                if preferred_provider:
                    for target in targets:
                        if preferred_provider in target:
                            resolved = target
                            logger.info(
                                f"Virtual model '{model_name}' resolved to '{resolved}' "
                                f"(matched preferred provider '{preferred_provider}')"
                            )
                            break
                    else:
                        resolved = targets[0]
                        logger.info(
                            f"Virtual model '{model_name}' resolved to '{resolved}' "
                            f"(preferred provider '{preferred_provider}' not found, using first target)"
                        )
                else:
                    # Could add weighted random selection here
                    resolved = targets[0]
                    logger.info(
                        f"Virtual model '{model_name}' resolved to '{resolved}' "
                        f"(no preference, using first of {len(targets)} targets)"
                    )
            elif isinstance(mapping, list):
                # List of targets directly
                if not mapping:
                    logger.error(f"Virtual model '{model_name}' has empty targets list")
                    return None
                resolved = mapping[0]
                logger.info(
                    f"Virtual model '{model_name}' resolved to '{resolved}' "
                    f"(list format, using first of {len(mapping)} targets)"
                )
            else:
                logger.error(
                    f"Invalid mapping type for virtual model '{model_name}': {type(mapping)}"
                )
                return None

            # Cache the resolution
            self.resolution_cache[cache_key] = resolved
            logger.debug(f"Cached resolution '{cache_key}' -> '{resolved}'")
            return resolved

        except Exception as e:
            logger.error(
                f"Error resolving virtual model '{model_name}': {e}", exc_info=True
            )
            return None

    def get_virtual_model_info(self, model_name: str) -> Optional[Dict[str, Any]]:
        """Get information about a virtual model configuration."""
        info = self.virtual_models.get(model_name)
        if info:
            logger.debug(f"Retrieved info for virtual model '{model_name}'")
        else:
            logger.debug(f"No virtual model info found for '{model_name}'")
        return info

    def list_virtual_models(self) -> List[str]:
        """List all registered virtual model names."""
        models = list(self.virtual_models.keys())
        logger.debug(f"Listing {len(models)} virtual models")
        return models

    def reload(self):
        """Reload virtual model configuration."""
        logger.info("Reloading virtual model configuration")
        self.resolution_cache.clear()
        self._load_virtual_models()
        logger.info("Virtual model configuration reloaded")


# Global resolver instance
_resolver: Optional[VirtualModelResolver] = None


def get_resolver() -> VirtualModelResolver:
    """Get or create the global virtual model resolver."""
    global _resolver
    if _resolver is None:
        logger.debug("Initializing global VirtualModelResolver")
        _resolver = VirtualModelResolver()
    return _resolver


def resolve_virtual_model(
    model_name: str, preferred_provider: Optional[str] = None
) -> str:
    """
    Convenience function to resolve a virtual model.
    Returns the original name if resolution fails or model is not virtual.
    """
    if not model_name:
        logger.warning("Empty model name provided to resolve_virtual_model")
        return model_name

    resolver = get_resolver()
    try:
        resolved = resolver.resolve(model_name, preferred_provider)
        if resolved is None:
            logger.warning(
                f"Resolution failed for '{model_name}', returning original"
            )
            return model_name
        if resolved != model_name:
            logger.info(f"Virtual model resolution: '{model_name}' -> '{resolved}'")
        return resolved
    except Exception as e:
        logger.error(f"Unexpected error in resolve_virtual_model: {e}", exc_info=True)
        return model_name
