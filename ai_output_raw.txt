# --- Security ---
api_key_header = APIKeyHeader(name="Authorization", auto_error=False)


async def verify_api_key(authorization: str = Depends(api_key_header)) -> bool:
    """Verify the API key from Authorization header."""
    if not proxy_api_key:
        return True
    
    if not authorization:
        raise HTTPException(status_code=401, detail="Missing authorization header")
    
    # Handle Bearer token format
    if authorization.startswith("Bearer "):
        token = authorization[7:]
    else:
        token = authorization
    
    if token != proxy_api_key:
        raise HTTPException(status_code=403, detail="Invalid API key")
    
    return True

@app.post("/v1/completions", response_model=CompletionResponse)
async def create_completion(
    request: CompletionRequest,
    authorized: bool = Depends(verify_api_key)
):
    """
    Create a legacy completion (non-chat).
    Supports both streaming and non-streaming responses.
    """
    try:
        router = get_router()
        
        # Convert request to dict for the router
        kwargs = request.dict(exclude_none=True)
        
        # Handle streaming response
        if request.stream:
            async def generate_stream():
                try:
                    response = await router.acompletion(**kwargs)
                    async for chunk in response:
                        yield f"data: {json.dumps(chunk)}\n\n"
                    yield "data: [DONE]\n\n"
                except Exception as e:
                    error_chunk = {
                        "error": {
                            "message": str(e),
                            "type": "internal_error",
                            "code": "500"
                        }
                    }
                    yield f"data: {json.dumps(error_chunk)}\n\n"
                    yield "data: [DONE]\n\n"
            
            return StreamingResponse(
                generate_stream(),
                media_type="text/event-stream",
                headers={
                    "Cache-Control": "no-cache",
                    "Connection": "keep-alive",
                    "X-Accel-Buffering": "no"
                }
            )
        
        # Non-streaming response
        response = await router.acompletion(**kwargs)
        
        # Handle different response types from litellm
        if hasattr(response, 'model_dump'):
            return response.model_dump()
        elif hasattr(response, 'dict'):
            return response.dict()
        else:
            return response
            
    except HTTPException:
        raise
    except Exception as e:
        raise HTTPException(
            status_code=500,
            detail=f"Completion error: {str(e)}"
        )
