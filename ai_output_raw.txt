"""
Health Checker Module for Uptime Monitoring

Provides comprehensive health monitoring, uptime tracking, and
Kubernetes-compatible health probe endpoints.
"""

import time
import asyncio
import threading
from datetime import datetime, timedelta
from typing import Dict, List, Optional, Callable, Any
from dataclasses import dataclass, field
from enum import Enum
import logging

logger = logging.getLogger(__name__)


class HealthStatus(Enum):
    """Health status enumeration."""
    HEALTHY = "healthy"
    DEGRADED = "degraded"
    UNHEALTHY = "unhealthy"
    UNKNOWN = "unknown"


@dataclass
class ProviderHealth:
    """Health metrics for a single provider."""
    name: str
    status: HealthStatus = HealthStatus.UNKNOWN
    last_check: Optional[datetime] = None
    last_success: Optional[datetime] = None
    total_requests: int = 0
    failed_requests: int = 0
    average_response_time: float = 0.0
    uptime_percentage: float = 100.0
    consecutive_failures: int = 0
    consecutive_successes: int = 0


@dataclass
class SystemHealth:
    """Overall system health status."""
    status: HealthStatus
    uptime_seconds: float
    start_time: datetime
    version: str = "1.0.0"
    checks: Dict[str, Any] = field(default_factory=dict)
    timestamp: datetime = field(default_factory=datetime.utcnow)


class HealthChecker:
    """
    Singleton health checker for uptime monitoring.
    
    Tracks:
    - Application uptime and availability
    - Per-provider health status and success rates
    - System readiness and liveness
    - Response times and error rates
    """
    
    _instance = None
    _lock = threading.Lock()
    
    def __new__(cls):
        if cls._instance is None:
            with cls._lock:
                if cls._instance is None:
                    cls._instance = super().__new__(cls)
                    cls._instance._initialized = False
        return cls._instance
    
    def __init__(self):
        if self._initialized:
            return
            
        self.start_time = datetime.utcnow()
        self.provider_health: Dict[str, ProviderHealth] = {}
        self._checks: Dict[str, Callable] = {}
        self._initialized = True
        self._monitoring_active = False
        self._version = "1.0.0"
        
    def register_health_check(self, name: str, check_func: Callable[[], bool]):
        """
        Register a custom health check function.
        
        Args:
            name: Identifier for this check
            check_func: Function returning True if healthy, False otherwise
        """
        self._checks[name] = check_func
        logger.info(f"Registered health check: {name}")
        
    def unregister_health_check(self, name: str):
        """Remove a registered health check."""
        if name in self._checks:
            del self._checks[name]
            
    def update_provider_status(
        self, 
        provider: str, 
        status: HealthStatus, 
        response_time: float = 0.0,
        error: Optional[str] = None
    ):
        """
        Update health status for a specific provider.
        
        Args:
            provider: Provider identifier
            status: HealthStatus enum value
            response_time: Response time in seconds
            error: Optional error message if status is UNHEALTHY
        """
        if provider not in self.provider_health:
            self.provider_health[provider] = ProviderHealth(name=provider)
            
        health = self.provider_health[provider]
        now = datetime.utcnow()
        health.last_check = now
        
        # Update request counters
        health.total_requests += 1
        
        if status == HealthStatus.HEALTHY:
            health.last_success = now
            health.consecutive_successes += 1
            health.consecutive_failures = 0
            
            # Update rolling average response time
            if health.total_requests > 1:
                health.average_response_time = (
                    (health.average_response_time * (health.total_requests - 1) + response_time)
                    / health.total_requests
                )
            else:
                health.average_response_time = response_time
        else:
            health.failed_requests += 1
            health.consecutive_failures += 1
            health.consecutive_successes = 0
            
        health.status = status
        
        # Calculate uptime percentage
        if health.total_requests > 0:
            health.uptime_percentage = (
                (health.total_requests - health.failed_requests) / health.total_requests
            ) * 100
            
    def get_liveness_status(self) -> Dict[str, Any]:
        """
        Kubernetes liveness probe endpoint.
        
        Returns 200 if the application is running, regardless of
        whether it can serve traffic.
        """
        uptime = (datetime.utcnow() - self.start_time).total_seconds()
        
        return {
            "status": "alive",
            "timestamp": datetime.utcnow().isoformat(),
            "uptime_seconds": uptime,
            "uptime_human": str(timedelta(seconds=int(uptime))),
            "version": self._version
        }
        
    def get_readiness_status(self) -> Dict[str, Any]:
        """
        Kubernetes readiness probe endpoint.
        
        Returns 200 only if the application is ready to accept traffic.
        Checks all registered health check functions.
        """
        checks = {}
        overall_ready = True
        
        for name, check_func in self._checks.items():
            try:
                if asyncio.iscoroutinefunction(check_func):
                    # Note: This should be called from async context
                    result = True  # Default for sync call
                else:
                    result = check_func()
                    
                checks[name] = {
                    "status": "pass" if result else "fail",
                    "timestamp": datetime.utcnow().isoformat()
                }
                if not result:
                    overall_ready = False
            except Exception as e:
                logger.error(f"Health check '{name}' failed: {e}")
                checks[name] = {
                    "status": "fail",
                    "error": str(e),
                    "timestamp": datetime.utcnow().isoformat()
                }
                overall_ready = False
                
        status_code = "ready" if overall_ready else "not_ready"
        
        return {
            "ready": overall_ready,
            "status": status_code,
            "checks": checks,
            "timestamp": datetime.utcnow().isoformat(),
            "uptime_seconds": (datetime.utcnow() - self.start_time).total_seconds()
        }
        
    def get_health_status(self) -> SystemHealth:
        """
        Comprehensive health status for monitoring dashboards.
        
        Includes all providers, system metrics, and health check results.
        """
        now = datetime.utcnow()
        uptime = (now - self.start_time).total_seconds()
        
        # Determine overall status
        if not self.provider_health:
            status = HealthStatus.HEALTHY
        else:
            unhealthy_count = sum(
                1 for p in self.provider_health.values() 
                if p.status == HealthStatus.UNHEALTHY
            )
            degraded_count = sum(
                1 for p in self.provider_health.values()
                if p.status == HealthStatus.DEGRADED
            )
            
            total = len(self.provider_health)
            if unhealthy_count == total:
                status = HealthStatus.UNHEALTHY
            elif unhealthy_count > 0 or degraded_count > 0:
                status = HealthStatus.DEGRADED
            else:
                status = HealthStatus.HEALTHY
                
        # Compile provider statuses
        provider_checks = {}
        for name, p in self.provider_health.items():
            provider_checks[name] = {
                "status": p.status.value,
                "uptime_percentage": round(p.uptime_percentage, 2),
                "last_check": p.last_check.isoformat() if p.last_check else None,
                "last_success": p.last_success.isoformat() if p.last_success else None,
                "average_response_time_ms": round(p.average_response_time * 1000, 2),
                "consecutive_failures": p.consecutive_failures,
                "consecutive_successes": p.consecutive_successes,
                "total_requests": p.total_requests,
                "failed_requests": p.failed_requests
            }
            
        return SystemHealth(
            status=status,
            uptime_seconds=uptime,
            start_time=self.start_time,
            version=self._version,
            checks={
                "providers": provider_checks,
                "registered_checks": list(self._checks.keys())
            },
            timestamp=now
        )
        
    def get_uptime_stats(self) -> Dict[str, Any]:
        """
        Detailed uptime statistics for reporting and SLA monitoring.
        """
        now = datetime.utcnow()
        uptime = (now - self.start_time).total_seconds()
        
        # Calculate aggregate statistics across all providers
        total_requests = sum(p.total_requests for p in self.provider_health.values())
        total_failures = sum(p.failed_requests for p in self.provider_health.values())
        overall_success_rate = (
            ((total_requests - total_failures) / total_requests * 100)
            if total_requests > 0 else 100.0
        )
        
        return {
            "system": {
                "start_time": self.start_time.isoformat(),
                "current_time": now.isoformat(),
                "uptime_seconds": uptime,
                "uptime_human": str(timedelta(seconds=int(uptime))),
                "version": self._version,
                "status": self.get_health_status().status.value
            },
            "aggregate": {
                "total_requests": total_requests,
                "total_failures": total_failures,
                "overall_success_rate": round(overall_success_rate, 2),
                "provider_count": len(self.provider_health),
                "healthy_providers": sum(
                    1 for p in self.provider_health.values() 
                    if p.status == HealthStatus.HEALTHY
                )
            },
            "providers": {
                name: {
                    "status": p.status.value,
                    "uptime_percentage": round(p.uptime_percentage, 2),
                    "total_requests": p.total_requests,
                    "failed_requests": p.failed_requests,
                    "success_rate": round(
                        ((p.total_requests - p.failed_requests) / p.total_requests * 100)
                        if p.total_requests > 0 else 100.0, 2
                    ),
                    "average_response_time_ms": round(p.average_response_time * 1000, 2),
                    "last_check": p.last_check.isoformat() if p.last_check else None,
                    "last_success": p.last_success.isoformat() if p.last_success else None,
                    "consecutive_failures": p.consecutive_failures
                }
                for name, p in self.provider_health.items()
            }
        }
        
    async def start_monitoring(self, interval: int = 60):
        """
        Start background health monitoring task.
        
        Periodically runs registered health checks and updates metrics.
        
        Args:
            interval: Seconds between health check runs
        """
        if self._monitoring_active:
            return
            
        self._monitoring_active = True
        logger.info(f"Starting health monitoring with {interval}s interval")
        
        while self._monitoring_active:
            try:
                await asyncio.sleep(interval)
                
                # Run async health checks
                for name, check_func in self._checks.items():
                    try:
                        if asyncio.iscoroutinefunction(check_func):
                            await check_func()
                    except Exception as e:
                        logger.error(f"Background health check '{name}' failed: {e}")
                        
            except asyncio.CancelledError:
                logger.info("Health monitoring cancelled")
                break
            except Exception as e:
                logger.error(f"Error in health monitoring loop: {e}")
                
    def stop_monitoring(self):
        """Stop the background monitoring task."""
        self._monitoring_active = False
        
    def reset_provider_stats(self, provider: Optional[str] = None):
        """
        Reset statistics for a provider or all providers.
        
        Args:
            provider: Provider name to reset, or None to reset all
        """
        if provider:
            if provider in self.provider_health:
                self.provider_health[provider] = ProviderHealth(name=provider)
        else:
            self.provider_health.clear()

"""
Status API Router for Health Checks and Uptime Monitoring

Provides REST endpoints for:
- Kubernetes liveness/readiness probes
- Health status dashboards
- Uptime statistics
- Provider availability metrics
"""

from fastapi import APIRouter, HTTPException, Depends
from fastapi.responses import JSONResponse
from typing import Dict, Any

from proxy_app.health_checker import HealthChecker, HealthStatus

# Create router
router = APIRouter(tags=["status"])

# Singleton instance
health_checker = HealthChecker()


@router.get("/health", summary="Comprehensive Health Check")
async def health_check() -> Dict[str, Any]:
    """
    Returns comprehensive health status including all providers and system metrics.
    
    Suitable for load balancer health checks and monitoring dashboards.
    Returns 503 if system is unhealthy, 200 if healthy or degraded.
    """
    health = health_checker.get_health_status()
    
    # Determine HTTP status code
    if health.status == HealthStatus.UNHEALTHY:
        status_code = 503
    else:
        status_code = 200
        
    return JSONResponse(
        status_code=status_code,
        content={
            "status": health.status.value,
            "uptime_seconds": round(health.uptime_seconds, 2),
            "uptime_human": str(timedelta(seconds=int(health.uptime_seconds))),
            "version": health.version,
            "timestamp": health.timestamp.isoformat(),
            "start_time": health.start_time.isoformat(),
            "checks": health.checks
        }
    )


@router.get("/health/live", summary="Liveness Probe")
async def liveness_probe() -> Dict[str, Any]:
    """
    Kubernetes liveness probe endpoint.
    
    Returns 200 if the application is running. This endpoint should be used
    by Kubernetes to determine if the container should be restarted.
    Always returns 200 unless the application is completely frozen.
    """
    status = health_checker.get_liveness_status()
    return JSONResponse(status_code=200, content=status)


@router.get("/health/ready", summary="Readiness Probe")
async def readiness_probe() -> Dict[str, Any]:
    """
    Kubernetes readiness probe endpoint.
    
    Returns 200 only if the application is ready to accept traffic.
    Returns 503 if critical dependencies are unavailable.
    """
    status = health_checker.get_readiness_status()
    status_code = 200 if status["ready"]
