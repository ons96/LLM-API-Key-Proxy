"""Dynamic Model Scoring Engine

Calculates real-time scores for models based on agentic coding performance,
tokens per second, and availability with configurable weights and thresholds.
"""

import time
import sqlite3
import threading
from typing import Dict, List, Optional, Tuple
from dataclasses import dataclass
from pathlib import Path
import logging
import yaml

logger = logging.getLogger(__name__)


@dataclass
class ModelScore:
    provider: str
    model: str
    agentic_score: float
    tps: float
    availability: float
    hallucination_rate: float
    web_search_capable: bool
    total_score: float
    meets_threshold: bool


class DynamicScoringEngine:
    """Calculates dynamic scores for virtual model fallback ordering.

    Models are ordered purely by performance benchmarks + TPS + hallucination.
    Rate limits and availability are NOT used for ordering — rate-limited
    providers are skipped at runtime instead.
    """

    CATEGORY_WEIGHTS = {
        "coding-elite": {
            "agentic": 0.75,
            "tps": 0.15,
            "hallucination_penalty": 0.10,
        },
        "coding-smart": {
            "agentic": 0.65,
            "tps": 0.20,
            "hallucination_penalty": 0.15,
        },
        "coding-fast": {
            "agentic": 0.30,
            "tps": 0.55,
            "hallucination_penalty": 0.15,
        },
        "chat-elite": {
            "intelligence": 0.80,
            "hallucination_penalty": 0.20,
        },
        "chat-smart": {
            "intelligence": 0.50,
            "tps": 0.30,
            "hallucination_penalty": 0.20,
        },
        "chat-fast": {
            "tps": 0.60,
            "intelligence": 0.20,
            "hallucination_penalty": 0.20,
        },
    }

    CODING_WEIGHTS = {
        "agentic": 0.75,
        "tps": 0.15,
        "hallucination_penalty": 0.10,
    }

    CHAT_WEIGHTS = {
        "intelligence": 0.60,
        "tps": 0.25,
        "hallucination_penalty": 0.15,
    }

    THRESHOLDS = {
        "coding-elite": 75.0,
        "coding-smart": 65.0,
        "coding-fast": 40.0,
        "chat-elite": 0.0,
        "chat-smart": 0.0,
        "chat-fast": 0.0,
    }

    def __init__(
        self,
        telemetry_manager=None,
        model_rankings_path: str = "config/model_rankings.yaml",
        scoring_config_path: str = "config/scoring_config.yaml",
    ):
        self.telemetry = telemetry_manager
        self.model_rankings_path = Path(model_rankings_path)
        self.scoring_config_path = Path(scoring_config_path)
        self._rankings_cache: Dict[str, Dict] = {}
        self._last_refresh = 0
        
        # Scoring result cache
        self._score_cache: Dict[str, Tuple[ModelScore, float]] = {}
        self._cache_lock = threading.RLock()
        self._cache_ttl_seconds = 300  # Default 5 minutes
        self._cache_max_entries = 10000
        self._cache_enabled = True
        
        self._load_scoring_config()

    def _load_scoring_config(self):
        """Load cache configuration from scoring config."""
        try:
            if self.scoring_config_path.exists():
                with open(self.scoring_config_path) as f:
                    config = yaml.safe_load(f)
                
                cache_config = config.get("scoring", {}).get("cache", {})
                self._cache_enabled = cache_config.get("enabled", True)
                self._cache_ttl_seconds = cache_config.get("ttl_seconds", 300)
                self._cache_max_entries = cache_config.get("max_entries", 10000)
                
                logger.info(f"Scoring cache configured: enabled={self._cache_enabled}, "
                           f"ttl={self._cache_ttl_seconds}s, max_entries={self._cache_max_entries}")
        except Exception as e:
            logger.warning(f"Failed to load scoring config, using defaults: {e}")

    def _get_cache_key(self, provider: str, model: str, category: str) -> str:
        """Generate cache key for provider/model/category combination."""
        return f"{provider}/{model}/{category}"

    def _get_cached_score(self, provider: str, model: str, category: str) -> Optional[ModelScore]:
        """Retrieve cached score if valid."""
        if not self._cache_enabled:
            return None
            
        key = self._get_cache_key(provider, model, category)
        
        with self._cache_lock:
            if key not in self._score_cache:
                return None
                
            score, timestamp = self._score_cache[key]
            age = time.time() - timestamp
            
            if age > self._cache_ttl_seconds:
                # Cache expired
                del self._score_cache[key]
                return None
                
            logger.debug(f"Cache hit for {key} (age={age:.1f}s)")
            return score

    def _set_cached_score(self, provider: str, model: str, category: str, score: ModelScore):
        """Store score in cache with current timestamp."""
        if not self._cache_enabled:
            return
            
        key = self._get_cache_key(provider, model, category)
        
        with self._cache_lock:
            # Enforce max entries limit (simple LRU eviction)
            if len(self._score_cache) >= self._cache_max_entries:
                # Remove oldest entry
                oldest_key = min(self._score_cache, key=lambda k: self._score_cache[k][1])
                del self._score_cache[oldest_key]
                logger.debug(f"Evicted oldest cache entry: {oldest_key}")
            
            self._score_cache[key] = (score, time.time())
            logger.debug(f"Cached score for {key}: {score.total_score:.2f}")

    def invalidate_cache(self, provider: Optional[str] = None, model: Optional[str] = None):
        """Invalidate cache entries.
        
        Args:
            provider: If specified, only invalidate entries for this provider
            model: If specified, only invalidate entries for this model
        """
        with self._cache_lock:
            if provider is None and model is None:
                count = len(self._score_cache)
                self._score_cache.clear()
                logger.info(f"Invalidated all {count} cached scores")
                return
            
            keys_to_delete = []
            for key in self._score_cache.keys():
                parts = key.split("/")
                if len(parts) >= 2:
                    key_provider, key_model = parts[0], parts[1]
                    if provider and key_provider == provider:
                        if model is None or key_model == model:
                            keys_to_delete.append(key)
                    elif model and key_model == model:
                        keys_to_delete.append(key)
            
            for key in keys_to_delete:
                del self._score_cache[key]
            
            if keys_to_delete:
                logger.info(f"Invalidated {len(keys_to_delete)} cache entries "
                           f"(provider={provider}, model={model})")

    def load_model_rankings(self) -> Dict[str, Dict]:
        """Load model rankings from YAML file."""
        try:
            with open(self.model_rankings_path) as f:
                data = yaml.safe_load(f)

            rankings = {}
            for model in data.get("models", []):
                model_id = model.get("id", "")
                if "/" in model_id:
                    provider, model_name = model_id.split("/", 1)
                    key = f"{provider}/{model_name}"
                    rankings[key] = model

            self._rankings_cache = rankings
            self._last_refresh = time.time()
            
            # Invalidate cache when rankings change
            self.invalidate_cache()
            
            return rankings

        except Exception as e:
            logger.error(f"Failed to load model rankings: {e}")
            return {}

    def get_model_score(self, provider: str, model: str) -> Optional[float]:
        """Get agentic coding score for a model."""
        if time.time() - self._last_refresh > 3600:
            self.load_model_rankings()

        key = f"{provider}/{model}"
        model_data = self._rankings_cache.get(key)

        if not model_data:
            for k, v in self._rankings_cache.items():
                if (
                    k.endswith(f"/{model}")
                    or v.get("name", "").lower() == model.lower()
                ):
                    model_data = v
                    break

        if model_data:
            scores = model_data.get("scores", {})
            return scores.get("agentic_coding", scores.get("swe_bench", 0.0))

        return None

    def get_hallucination_rate(self, provider: str, model: str) -> float:
        """Get hallucination rate for a model (0-100, lower is better)."""
        if time.time() - self._last_refresh > 3600:
            self.load_model_rankings()

        key = f"{provider}/{model}"
        model_data = self._rankings_cache.get(key)

        if not model_data:
            for k, v in self._rankings_cache.items():
                if (
                    k.endswith(f"/{model}")
                    or v.get("name", "").lower() == model.lower()
                ):
                    model_data = v
                    break

        if model_data:
            scores = model_data.get("scores", {})
            return scores.get("hallucination_rate", 10.0)

        return 10.0

    def get_web_search_capable(self, provider: str, model: str) -> bool:
        """Check if model supports web search."""
        if time.time() - self._last_refresh > 3600:
            self.load_model_rankings()

        key = f"{provider}/{model}"
        model_data = self._rankings_cache.get(key)

        if not model_data:
            for k, v in self._rankings_cache.items():
                if (
                    k.endswith(f"/{model}")
                    or v.get("name", "").lower() == model.lower()
                ):
                    model_data = v
                    break

        if model_data:
            capabilities = model_data.get("capabilities", {})
            return capabilities.get("web_search_capable", False)

        return False

    def get_tps_estimate(self, provider: str, model: str) -> float:
        """Get TPS from telemetry or return default estimate."""
        if not self.telemetry:
            return 100.0

        try:
            conn = sqlite3.connect(self.telemetry.db_path)
            cursor = conn.cursor()

            # Query average TPS from recent requests (last hour)
            cursor.execute("""
                SELECT AVG(tokens_generated / NULLIF(generation_time_seconds, 0)) as avg_tps
                FROM requests 
                WHERE provider = ? 
                AND model = ? 
                AND timestamp > datetime('now', '-1 hour')
                AND success = 1
            """, (provider, model))

            result = cursor.fetchone()
            conn.close()

            if result and result[0]:
                return float(result[0])
            return 100.0  # Default TPS

        except Exception as e:
            logger.debug(f"Failed to get TPS for {provider}/{model}: {e}")
            return 100.0

    def calculate_dynamic_score(
        self, 
        provider: str, 
        model: str, 
        category: str = "coding-elite",
        force_refresh: bool = False
    ) -> Optional[ModelScore]:
        """Calculate dynamic score for a provider/model with caching.
        
        Args:
            provider: Provider name
            model: Model name
            category: Scoring category (coding-elite, coding-smart, etc.)
            force_refresh: If True, bypass cache and recalculate
            
        Returns:
            ModelScore object or None if calculation fails
        """
        # Check cache first
        if not force_refresh:
            cached = self._get_cached_score(provider, model, category)
            if cached:
                return cached
        
        # Gather metrics
        agentic = self.get_model_score(provider, model) or 0.0
        tps = self.get_tps_estimate(provider, model)
        hallucination = self.get_hallucination_rate(provider, model)
        web_search = self.get_web_search_capable(provider, model)
        
        # Get weights for category
        weights = self.CATEGORY_WEIGHTS.get(category, self.CODING_WEIGHTS)
        
        # Calculate normalized TPS (0-100 scale, assuming max 3000 TPS)
        normalized_tps = min(tps / 30.0, 100.0)  # 3000 TPS = 100 points
        
        # Calculate score based on category type
        if category.startswith("coding"):
            agentic_weight = weights.get("agentic", 0.75)
            tps_weight = weights.get("tps", 0.15)
            hall_penalty_weight = weights.get("hallucination_penalty", 0.10)
            
            # Hallucination factor: 0-100 scale inverted (100 = no hallucination)
            hall_factor = max(0, 100 - hallucination * 10) / 100.0
            
            # Base score from agentic and TPS
            base_score = agentic * agentic_weight + normalized_tps * tps_weight
            
            # Apply hallucination penalty
            penalty = base_score * hall_penalty_weight * (1 - hall_factor)
            total_score = base_score - penalty
            
        elif category.startswith("chat"):
            intelligence_weight = weights.get("intelligence", 0.60)
            tps_weight = weights.get("tps", 0.25)
            hall_penalty_weight = weights.get("hallucination_penalty", 0.15)
            
            # For chat, use agentic score as proxy for intelligence if available
            intelligence = agentic if agentic > 0 else 50.0
            
            # Normalize intelligence to 0-100
            normalized_intel = min(intelligence, 100.0)
            
            # Hallucination factor
            hall_factor = max(0, 100 - hallucination * 10) / 100.0
            
            base_score = normalized_intel * intelligence_weight + normalized_tps * tps_weight
            penalty = base_score * hall_penalty_weight * (1 - hall_factor)
            total_score = base_score - penalty
        else:
            # Default fallback
            total_score = agentic * 0.5 + normalized_tps * 0.5
        
        # Check threshold
        threshold = self.THRESHOLDS.get(category, 0.0)
        meets_threshold = total_score >= threshold
        
        score = ModelScore(
            provider=provider,
            model=model,
            agentic_score=agentic,
            tps=tps,
            availability=1.0,  # Determined at runtime, not cached
            hallucination_rate=hallucination,
            web_search_capable=web_search,
            total_score=total_score,
            meets_threshold=meets_threshold
        )
        
        # Cache the result
        self._set_cached_score(provider, model, category, score)
        
        return score

    def get_cache_stats(self) -> Dict:
        """Get current cache statistics."""
        with self._cache_lock:
            now = time.time()
            total_entries = len(self._score_cache)
            if total_entries == 0:
                return {
                    "enabled": self._cache_enabled,
                    "total_entries": 0,
                    "hit_rate": 0.0,
                    "avg_age_seconds": 0.0
                }
            
            ages = [now - ts for _, ts in self._score_cache.values()]
            avg_age = sum(ages) / len(ages)
            
            return {
                "enabled": self._cache_enabled,
                "total_entries": total_entries,
                "ttl_seconds": self._cache_ttl_seconds,
                "max_entries": self._cache_max_entries,
                "avg_age_seconds": round(avg_age, 2),
                "oldest_entry_seconds": round(max(ages), 2),
                "newest_entry_seconds": round(min(ages), 2)
            }

    def clear_expired_cache(self) -> int:
        """Clear expired entries from cache. Returns number of entries removed."""
        if not self._cache_enabled:
            return 0
            
        now = time.time()
        expired_keys = []
        
        with self._cache_lock:
            for key, (_, timestamp) in self._score_cache.items():
                if now - timestamp > self._cache_ttl_seconds:
                    expired_keys.append(key)
            
            for key in expired_keys:
                del self._score_cache[key]
        
        if expired_keys:
            logger.debug(f"Cleared {len(expired_keys)} expired cache entries")
        
        return len(expired_keys)
