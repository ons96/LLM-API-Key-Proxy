class TestGroqAdapterInitialization:
    """Tests for Groq adapter creation and setup."""
    
    def test_adapter_creation_with_valid_key(self, mock_groq_api_key: str):
        """Test successful adapter creation with valid API key."""
        factory = ProviderAdapterFactory()
        adapter = factory.create_adapter("groq", mock_groq_api_key)
        
        assert adapter is not None
        assert hasattr(adapter, 'provider_name')
        assert adapter.provider_name == "groq"
    
    def test_adapter_creation_without_key_raises_error(self):
        """Test that adapter creation fails appropriately without API key."""
        factory = ProviderAdapterFactory()
        
        with pytest.raises((ValueError, HTTPException, Exception)):
            factory.create_adapter("groq", None)
    
    def test_adapter_has_required_interface(self, mock_groq_api_key: str):
        """Test that adapter implements required provider interface."""
        factory = ProviderAdapterFactory()
        adapter = factory.create_adapter("groq", mock_groq_api_key)
        
        # Required methods for provider integration
        required_methods = ['chat_completions', 'get_capabilities']
        for method in required_methods:
            assert hasattr(adapter, method), f"Groq adapter missing required method: {method}"
    
    def test_groq_adapter_from_environment(self, mock_groq_api_key: str):
        """Test that adapter can be initialized from environment variables."""
        with patch.dict(os.environ, {"GROQ_API_KEY": mock_groq_api_key}):
            factory = ProviderAdapterFactory()
            adapter = factory.create_adapter("groq", os.getenv("GROQ_API_KEY"))
            assert adapter is not None
