#!/usr/bin/env python3
"""
Provider Status API Module

FastAPI endpoints for accessing provider health status information.
"""

from fastapi import APIRouter, Depends, HTTPException, Request, status
from fastapi.responses import JSONResponse, StreamingResponse
from pydantic import BaseModel, Field
from typing import Dict, Any, List, Optional
from datetime import datetime
from enum import Enum
import csv
import io
import logging

from rotator_library.provider_status_tracker import ProviderStatusTracker

# Module-level status tracker instance
status_tracker = ProviderStatusTracker()

# Configure logging
logger = logging.getLogger(__name__)

# Create router for status API endpoints
router = APIRouter(prefix="/api/providers", tags=["provider-status"])


# ============================================================================
# Pydantic Models for API Documentation
# ============================================================================

class ProviderHealthStatus(str, Enum):
    """Enumeration of possible provider health statuses."""
    HEALTHY = "healthy"
    DEGRADED = "degraded"
    UNHEALTHY = "unhealthy"
    UNKNOWN = "unknown"


class ProviderStatusDetail(BaseModel):
    """Detailed status information for a single provider."""
    provider_name: str = Field(..., description="Unique identifier for the provider")
    status: ProviderHealthStatus = Field(..., description="Current health status of the provider")
    latency_ms: Optional[float] = Field(None, description="Average response latency in milliseconds")
    success_rate: Optional[float] = Field(None, ge=0, le=1, description="Success rate (0.0 to 1.0)")
    last_check: Optional[datetime] = Field(None, description="Timestamp of the last health check")
    error_count: int = Field(0, ge=0, description="Number of consecutive errors")
    request_count: int = Field(0, ge=0, description="Total requests processed")
    active: bool = Field(True, description="Whether the provider is currently active")
    
    class Config:
        json_schema_extra = {
            "example": {
                "provider_name": "openai",
                "status": "healthy",
                "latency_ms": 245.5,
                "success_rate": 0.98,
                "last_check": "2024-01-15T10:30:00Z",
                "error_count": 0,
                "request_count": 1500,
                "active": True
            }
        }


class ProviderStatusResponse(BaseModel):
    """Response model for provider status listing."""
    timestamp: datetime = Field(..., description="When the status snapshot was generated")
    providers: Dict[str, ProviderStatusDetail] = Field(..., description="Map of provider names to their status details")
    
    class Config:
        json_schema_extra = {
            "example": {
                "timestamp": "2024-01-15T10:30:00Z",
                "providers": {
                    "openai": {
                        "provider_name": "openai",
                        "status": "healthy",
                        "latency_ms": 245.5,
                        "success_rate": 0.98,
                        "last_check": "2024-01-15T10:30:00Z",
                        "error_count": 0,
                        "request_count": 1500,
                        "active": True
                    }
                }
            }
        }


class BestProviderResponse(BaseModel):
    """Response model for best provider recommendation."""
    provider_name: str = Field(..., description="Name of the recommended best provider")
    score: float = Field(..., description="Health score (higher is better)")
    latency_ms: Optional[float] = Field(None, description="Current latency in milliseconds")
    alternatives: List[str] = Field(default=[], description="List of alternative provider names")
    reason: str = Field(..., description="Reason for this recommendation")
    
    class Config:
        json_schema_extra = {
            "example": {
                "provider_name": "anthropic",
                "score": 0.95,
                "latency_ms": 180.0,
                "alternatives": ["openai", "cohere"],
                "reason": "Lowest latency with high success rate"
            }
        }


class ProviderHistoryEntry(BaseModel):
    """Single historical data point for a provider."""
    timestamp: datetime = Field(..., description="When this measurement was taken")
    status: ProviderHealthStatus = Field(..., description="Status at this time")
    latency_ms: Optional[float] = Field(None, description="Latency measurement")
    success_rate: Optional[float] = Field(None, ge=0, le=1, description="Success rate at this time")
    error_count: int = Field(0, description="Error count at this time")
    
    class Config:
        json_schema_extra = {
            "example": {
                "timestamp": "2024-01-15T09:00:00Z",
                "status": "healthy",
                "latency_ms": 250.0,
                "success_rate": 0.97,
                "error_count": 0
            }
        }


class ErrorResponse(BaseModel):
    """Standard error response model."""
    detail: str = Field(..., description="Error message describing what went wrong")
    
    class Config:
        json_schema_extra = {
            "example": {
                "detail": "Provider 'unknown_provider' not found"
            }
        }


# ============================================================================
# Dependencies
# ============================================================================

def get_status_tracker(request: Request) -> ProviderStatusTracker:
    """Dependency to get the provider status tracker from app state."""
    if not hasattr(request.app.state, "provider_status_tracker"):
        raise HTTPException(
            status_code=500, detail="Provider status tracker not initialized"
        )
    return request.app.state.provider_status_tracker


# ============================================================================
# API Endpoints
# ============================================================================

@router.get(
    "/status",
    response_model=ProviderStatusResponse,
    summary="Get all provider statuses",
    description="Retrieve a complete snapshot of the current health status for all configured providers, including latency, success rates, and error counts.",
    responses={
        500: {"model": ErrorResponse, "description": "Internal server error when retrieving status"}
    }
)
async def get_provider_status(
    request: Request, tracker: ProviderStatusTracker = Depends(get_status_tracker)
) -> Dict[str, Any]:
    """
    Get current status snapshot for all providers.

    Returns:
        JSON with timestamp and provider status details including health, latency, and reliability metrics.
    """
    try:
        status = tracker.get_current_status()
        return status
    except Exception as e:
        logger.error(f"Failed to get provider status: {e}")
        raise HTTPException(
            status_code=500, detail=f"Failed to get provider status: {e}"
        )


@router.get(
    "/status/{provider_name}",
    response_model=ProviderStatusDetail,
    summary="Get single provider status",
    description="Retrieve detailed health status information for a specific provider by name.",
    responses={
        404: {"model": ErrorResponse, "description": "Provider not found"},
        500: {"model": ErrorResponse, "description": "Internal server error"}
    }
)
async def get_single_provider_status(
    request: Request,
    provider_name: str,
    tracker: ProviderStatusTracker = Depends(get_status_tracker),
) -> Dict[str, Any]:
    """
    Get detailed status for a single provider.

    Args:
        provider_name: Name of the provider to query (e.g., 'openai', 'anthropic')

    Returns:
        JSON object containing provider health metrics, performance statistics, and current state.
    """
    try:
        current_status = tracker.get_current_status()

        if provider_name not in current_status["providers"]:
            raise HTTPException(
                status_code=404, detail=f"Provider '{provider_name}' not found"
            )

        provider_data = current_status["providers"][provider_name]

        # Add provider name to response
        provider_data["provider_name"] = provider_name

        return provider_data

    except HTTPException:
        raise
    except Exception as e:
        logger.error(f"Failed to get status for {provider_name}: {e}")
        raise HTTPException(
            status_code=500, detail=f"Failed to get provider status: {e}"
        )


@router.get(
    "/best",
    response_model=BestProviderResponse,
    summary="Get best available provider",
    description="Returns the healthiest and fastest provider currently available, along with alternative options and scoring rationale.",
    responses={
        500: {"model": ErrorResponse, "description": "Internal server error when calculating best provider"}
    }
)
async def get_best_provider(
    request: Request, tracker: ProviderStatusTracker = Depends(get_status_tracker)
) -> Dict[str, Any]:
    """
    Get the healthiest/fastest provider recommendation.

    Returns:
        JSON containing the recommended provider name, health score, latency metrics, 
        and a list of alternative providers sorted by preference.
    """
    try:
        best_provider = tracker.get_best_provider()
        return best_provider
    except Exception as e:
        logger.error(f"Failed to get best provider: {e}")
        raise HTTPException(status_code=500, detail=f"Failed to get best provider: {e}")


@router.get(
    "/history",
    response_model=Dict[str, List[ProviderHistoryEntry]],
    summary="Get historical data for all providers",
    description="Retrieve time-series historical data for all providers over a specified time window. Useful for trend analysis and debugging.",
    responses={
        500: {"model": ErrorResponse, "description": "Internal server error retrieving history"}
    }
)
async def get_provider_history(
    request: Request,
    hours: int = Field(24, ge=1, le=168, description="Time window in hours (1-168)"),
    tracker: ProviderStatusTracker = Depends(get_status_tracker),
) -> Dict[str, List[Dict[str, Any]]]:
    """
    Get historical performance data for all providers.

    Args:
        hours: Time window in hours to retrieve (default: 24, max: 168)

    Returns:
        Dictionary mapping provider names to lists of historical measurements including
        timestamps, latency, success rates, and status changes.
    """
    try:
        history = tracker.get_all_history(hours)
        return history
    except Exception as e:
        logger.error(f"Failed to get provider history: {e}")
        raise HTTPException(
            status_code=500, detail=f"Failed to get provider history: {e}"
        )


@router.get(
    "/history/{provider_name}",
    response_model=List[ProviderHistoryEntry],
    summary="Get historical data for single provider",
    description="Retrieve time-series historical data for a specific provider, useful for analyzing performance trends and reliability over time.",
    responses={
        404: {"model": ErrorResponse, "description": "No history found for provider"},
        500: {"model": ErrorResponse, "description": "Internal server error"}
    }
)
async def get_single_provider_history(
    request: Request,
    provider_name: str,
    hours: int = Field(24, ge=1, le=168, description="Time window in hours (1-168)"),
    tracker: ProviderStatusTracker = Depends(get_status_tracker),
) -> List[Dict[str, Any]]:
    """
    Get historical performance data for a single provider.

    Args:
        provider_name: Name of the provider to query (e.g., 'openai', 'anthropic')
        hours: Time window in hours (default: 24, max: 168)

    Returns:
        List of historical measurements sorted by timestamp, including latency,
        success rates, and status at each measurement point.
    """
    try:
        history = tracker.get_provider_history(provider_name, hours)

        if not history:
            raise HTTPException(
                status_code=404,
                detail=f"No history found for provider '{provider_name}'",
            )

        return history

    except HTTPException:
        raise
    except Exception as e:
        logger.error(f"Failed to get history for {provider_name}: {e}")
        raise HTTPException(
            status_code=500, detail=f"Failed to get provider history: {e}"
        )


@router.get(
    "/export/csv",
    summary="Export provider status to CSV",
    description="Export the current status of all providers in CSV format for external analysis and reporting. Returns a downloadable CSV file with provider metrics.",
    responses={
        200: {
            "content": {"text/csv": {}},
            "description": "CSV file containing provider status data"
        },
        500: {"model": ErrorResponse, "description": "Internal server error during export"}
    }
)
async def export_provider_status_csv(
    request: Request, tracker: ProviderStatusTracker = Depends(get_status_tracker)
) -> StreamingResponse:
    """
    Export current provider status to CSV format.

    Returns:
        CSV file with columns: provider_name, status, latency_ms, success_rate, 
        last_check, error_count, request_count, active.
    """
    try:
        csv_data = tracker.export_to_csv()

        # Create streaming response
        stream = io.StringIO(csv_data)

        return StreamingResponse(
            iter([stream.getvalue()]),
            media_type="text/csv",
            headers={
                "Content-Disposition": "attachment; filename=provider_status.csv",
                "Content-Type": "text/csv",
            },
        )
    except Exception as e:
        logger.error(f"Failed to export provider status to CSV: {e}")
        raise HTTPException(
            status_code=500, detail=f"Failed to export provider status: {e}"
        )
