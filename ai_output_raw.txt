from .error_handler import ClassifiedError, NoAvailableKeysError, mask_credential

lib_logger = logging.getLogger("rotator_library")
lib_logger.propagate = False
if not lib_logger.handlers:
    lib_logger.addHandler(logging.NullHandler())


class UsageManager:
    """
    Manages usage statistics and cooldowns for API keys with asyncio-safe locking,
    asynchronous file I/O, lazy-loading mechanism, and weighted random credential rotation.

    The credential rotation strategy can be configured via the `rotation_tolerance` parameter:

    - **tolerance = 0.0**: Deterministic least-used selection. The credential with
      the lowest usage count is always selected. This provides predictable, perfectly balanced
      load distribution but may be vulnerable to fingerprinting.

    - **tolerance = 2.0 - 4.0 (default, recommended)**: Balanced weighted randomness. Credentials are selected
      randomly with weights biased toward less-used ones. Credentials within 2 uses of the
      maximum can still be selected with reasonable probability. This provides security through
      unpredictability while maintaining good load balance.

    - **tolerance = 5.0+**: High randomness. Even heavily-used credentials have significant
      selection probability. Useful for stress testing or maximum unpredictability, but may
      result in less balanced load distribution.

    The weight formula is: `weight = (max_usage - credential_usage) + tolerance + 1`

    This ensures lower-usage credentials are preferred while tolerance controls how much
      randomness is introduced into the selection process.

    Additionally, providers can specify a rotation mode:
    - "balanced" (default): Rotate credentials to distribute load evenly
    - "sequential": Use one credential until exhausted (preserves caching)

    **Streaming Usage Tracking:**

    The UsageManager tracks streaming API usage separately from standard request usage to
    provide accurate load balancing for long-running streaming connections:

    - **stream_starts**: Count of streaming requests initiated (per credential)
    - **active_streams**: Currently active streaming connections (for concurrency management)
    - **stream_tokens**: Total tokens processed through streams (when available)
    - **stream_duration_ms**: Cumulative duration of streaming sessions

    Streaming usage is tracked via:
    - `record_stream_start(credential)`: Call when a streaming connection begins
    - `record_stream_end(credential, tokens=None)`: Call when a streaming connection ends

    The rotation algorithm considers active stream counts as part of the usage weight
    calculation when `rotation_tolerance` > 0, ensuring credentials with many active
    streams are less likely to be selected for new requests.
    """

    def __init__(
        self,
        file_path: Optional[Union[str, Path]] = None,
        daily_reset_time_utc: Optional[str] = "03:00",
        rotation_tolerance: float = 0.0,
        provider_rotation_modes: Optional[Dict[str, str]] = None,
        provider_plugins: Optional[Dict[str, Any]] = None,
        priority_multipliers: Optional[Dict[str, Dict[int, int]]] = None,
        priority_multipliers_by_mode: Optional[
            Dict[str, Dict[str, Dict[int, int]]]
        ] = None,
        sequential_fallback_multipliers: Optional[Dict[str, int]] = None,
    ):
        """
        Initialize the UsageManager.

        Args:
            file_path: Path to the usage data JSON file. If None, uses get_data_file("key_usage.json").
                       Can be absolute Path, relative Path, or string.
            daily_reset_time_utc: Time in UTC when daily stats should reset (HH:MM format)
            rotation_tolerance: Tolerance for weighted random credential rotation.
                - 0.0: Deterministic, least-used credential always selected
                - tolerance = 2.0 - 4.0 (default, recommended): Balanced randomness, can pick credentials within 2 uses of max
                - 5.0+: High randomness, more unpredictable selection patterns
            provider_rotation_modes: Dict mapping provider names to rotation modes.
                - "balanced": Rotate credentials to distribute load evenly (default)
                - "sequential": Use one credential until exhausted (preserves caching)
            provider_plugins: Dict mapping provider names to provider plugin instances.
                Used for per-provider usage reset configuration (window durations, field names).
            priority_multipliers: Dict mapping provider -> priority -> multiplier.
                Universal multipliers that apply regardless of rotation mode.
                Example: {"antigravity": {1: 5, 2: 3}}
            priority_multipliers_by_mode: Dict mapping provider -> mode -> priority -> multiplier.
                Mode-specific overrides. Example: {"antigravity": {"balanced": {3: 1}}}
            sequential_fallback_multipliers: Dict mapping provider -> fallback multiplier.
                Used in sequential mode when priority not in priority_multipliers.
                Example: {"antigravity": 2}
        """
        # Resolve file_path - use default if not provided
        if file_path is None:
            self.file_path = str(get_data_file("key_usage.json"))
        elif isinstance(file_path, Path):
            self.file_path = str(file_path)
        else:
            # String path - could be relative or absolute
            self.file_path = file_path
        self.rotation_tolerance = rotation_tolerance
        self.provider_rotation_modes = provider_rotation_modes or {}
        self.provider_plugins = provider_plugins or PROVIDER_PLUGINS
        self.priority_multipliers = priority_multipliers or {}
        self.priority_multipliers_by_mode = priority_multipliers_by_mode or {}
        self.sequential_fallback_multipliers = sequential_fallback_multipliers or {}
        self._provider_instances: Dict[str, Any] = {}  # Cache for provider instances
        self.key_states: Dict[str, Dict[str, Any]] = {}

        self._data_lock = asyncio.Lock()
        self._usage_data: Optional[Dict] = None
        self._initialized = asyncio.Event()
        self._init_lock = asyncio.Lock()

        self._timeout_lock = asyncio.Lock()
        self._claimed_on_timeout: Set[str] = set()

        # Resilient writer for usage data persistence
        self._state_writer = ResilientStateWriter(self.file_path, lib_logger)

        if daily_reset_time_utc:
            hour, minute = map(int, daily_reset_time_utc.split(":"))
            self.daily_reset_time_utc = dt_time(
                hour=hour, minute=minute, tzinfo=timezone.utc
            )
        else:
            self.daily_reset_time_utc = None

        # Track active streams in memory (not persisted)
        self._active_streams: Dict[str, int] = {}
        self._stream_lock = asyncio.Lock()

    def _get_rotation_mode(self, provider: str) -> str:
        """
        Get the rotation mode for a provider.

        Args:
            provider: Provider name (e.g., "antigravity", "gemini_cli")

        Returns:
            "balanced" or "sequential"
        """
        return self.provider_rotation_modes.get(provider, "balanced")

    def _get_priority_multiplier(
        self, provider: str, priority: int, rotation_mode: str
    ) -> int:
        """
        Get the concurrency multiplier for a provider/priority/mode combination.

        Lookup order:
        1. Mode-specific tier override: priority_multipliers_by_mode[provider][mode][priority]
        2. Universal tier multiplier: priority_multipliers[provider][priority]
        3. Sequential fallback (if mode is sequential): sequential_fallback_multipliers[provider]
        4. Global default: 1 (no multiplier effect)

        Args:
            provider: Provider name (e.g., "antigravity")
            priority: Priority level (1 = highest priority)
            rotation_mode: Current rotation mode ("sequential" or "balanced")

        Returns:
            Multiplier value
        """
        provider_lower = provider.lower()

        # 1. Check mode-specific override
        if provider_lower in self.priority_multipliers_by_mode:
            mode_multipliers = self.priority_multipliers_by_mode[provider_lower]
            if rotation_mode in mode_multipliers:
                if priority in mode_multipliers[rotation_mode]:
                    return mode_multipliers[rotation_mode][priority]

        # 2. Check universal tier multiplier
        if provider_lower in self.priority_multipliers:
            if priority in self.priority_multipliers[provider_lower]:
                return self.priority_multipliers[provider_lower][priority]

        # 3. Sequential fallback (only for sequential mode)
        if rotation_mode == "sequential":
            if provider_lower in self.sequential_fallback_multipliers:
                return self.sequential_fallback_multipliers[provider_lower]

        # 4. Global default
        return 1

    def _get_provider_from_credential(self, credential: str) -> Optional[str]:
        """
        Extract provider name from credential path or identifier.

        Supports multiple credential formats:
        - OAuth: "oauth_creds/antigravity_oauth_15.json" -> "antigravity"
        - OAuth: "C:\\...\\oauth_creds\\gemini_cli_oauth_1.json" -> "gemini_cli"
        - API key style: stored with provider prefix metadata

        Args:
            credential: The credential identifier (path or key)

        Returns:
            Provider name string or None if cannot be determined
        """
        import re

        # Normalize path separators
        normalized = credential.replace("\\", "/")

        # Try to extract from oauth path pattern: .../provider_oauth_N.json
        oauth_match = re.search(r'([^/]+)_oauth_\d+\.json$', normalized)
        if oauth_match:
            return oauth_match.group(1).lower()

        # Try to extract from api_key pattern: provider_...
        api_key_match = re.search(r'^([a-zA-Z0-9_]+)_', normalized)
        if api_key_match:
            return api_key_match.group(1).lower()

        return None

    async def _ensure_initialized(self):
        """Ensure usage data is loaded from disk."""
        if self._initialized.is_set():
            return

        async with self._init_lock:
            if self._initialized.is_set():
                return

            try:
                await self._load_usage_data()
            except Exception as e:
                lib_logger.warning(f"Failed to load usage data: {e}. Starting fresh.")
                self._usage_data = self._create_default_usage_data()
            finally:
                self._initialized.set()

    def _create_default_usage_data(self) -> Dict:
        """Create default empty usage data structure."""
        return {
            "credentials": {},
            "daily_stats": {},
            "last_reset": datetime.now(timezone.utc).isoformat(),
            "version": "2.0",  # Added for streaming tracking support
        }

    async def _load_usage_data(self):
        """Load usage data from disk asynchronously."""
        try:
            async with aiofiles.open(self.file_path, 'r') as f:
                content = await f.read()
                self._usage_data = json.loads(content)
                
                # Migrate old format if necessary
                if self._usage_data.get("version") != "2.0":
                    await self._migrate_usage_data()
                    
        except FileNotFoundError:
            self._usage_data = self._create_default_usage_data()
            await self._persist_usage_data()

    async def _migrate_usage_data(self):
        """Migrate old usage data format to new version with streaming support."""
        self._usage_data["version"] = "2.0"
        
        # Ensure all credentials have streaming fields
        for cred_id, cred_data in self._usage_data.get("credentials", {}).items():
            if "streaming" not in cred_data:
                cred_data["streaming"] = {
                    "stream_starts": 0,
                    "stream_tokens": 0,
                    "stream_duration_ms": 0,
                    "last_stream_start": None,
                }
        
        await self._persist_usage_data()
        lib_logger.info("Migrated usage data to version 2.0 with streaming support")

    async def _persist_usage_data(self):
        """Persist usage data to disk asynchronously."""
        if self._usage_data is None:
            return
            
        await self._state_writer.write(self._usage_data)

    async def record_stream_start(self, credential: str) -> None:
        """
        Record the start of a streaming session for the given credential.
        
        This increments the stream_starts counter and tracks the start time.
        Active stream count is maintained in memory for load balancing.
        
        Args:
            credential: The credential identifier (path or key)
        """
        await self._ensure_initialized()
        cred_id = mask_credential(credential)
        
        async with self._data_lock:
            if cred_id not in self._usage_data["credentials"]:
                self._usage_data["credentials"][cred_id] = self._create_default_credential_stats()
            
            cred_stats = self._usage_data["credentials"][cred_id]
            
            # Initialize streaming section if missing
            if "streaming" not in cred_stats:
                cred_stats["streaming"] = {
                    "stream_starts": 0,
                    "stream_tokens": 0,
                    "stream_duration_ms": 0,
                    "last_stream_start": None,
                }
            
            cred_stats["streaming"]["stream_starts"] += 1
            cred_stats["streaming"]["last_stream_start"] = datetime.now(timezone.utc).isoformat()
            cred_stats["last_used"] = datetime.now(timezone.utc).isoformat()
            
            # Persist usage data
            await self._persist_usage_data()
        
        # Update in-memory active stream count
        async with self._stream_lock:
            self._active_streams[cred_id] = self._active_streams.get(cred_id, 0) + 1
            
        lib_logger.debug(f"Stream started for credential {cred_id}. Active streams: {self._active_streams.get(cred_id, 0)}")

    async def record_stream_end(self, credential: str, tokens: Optional[int] = None, duration_ms: Optional[int] = None) -> None:
        """
        Record the end of a streaming session for the given credential.
        
        Args:
            credential: The credential identifier (path or key)
            tokens: Optional number of tokens processed in the stream
            duration_ms: Optional duration of the stream in milliseconds
        """
        cred_id = mask_credential(credential)
        
        # Update in-memory active stream count
        async with self._stream_lock:
            if cred_id in self._active_streams and self._active_streams[cred_id] > 0:
                self._active_streams[cred_id] -= 1
                active_count = self._active_streams[cred_id]
            else:
                active_count = 0
                lib_logger.warning(f"Stream end recorded for {cred_id} but no active streams tracked")
        
        # Update persistent stats
        async with self._data_lock:
            if cred_id in self._usage_data.get("credentials", {}):
                cred_stats = self._usage_data["credentials"][cred_id]
                
                if "streaming" in cred_stats:
                    if tokens is not None:
                        cred_stats["streaming"]["stream_tokens"] = cred_stats["streaming"].get("stream_tokens", 0) + tokens
                    if duration_ms is not None:
                        cred_stats["streaming"]["stream_duration_ms"] = cred_stats["streaming"].get("stream_duration_ms", 0) + duration_ms
                    
                    cred_stats["streaming"]["last_stream_end"] = datetime.now(timezone.utc).isoformat()
                    
                    await self._persist_usage_data()
        
        lib_logger.debug(f"Stream ended for credential {cred_id}. Tokens: {tokens}, Duration: {duration_ms}ms, Remaining active: {active_count}")

    def _create_default_credential_stats(self) -> Dict:
        """Create default stats structure for a new credential."""
        return {
            "usage_count": 0,
            "error_count": 0,
            "last_used": None,
            "first_used": datetime.now(timezone.utc).isoformat(),
            "streaming": {
                "stream_starts": 0,
                "stream_tokens": 0,
                "stream_duration_ms": 0,
                "last_stream_start": None,
                "last_stream_end": None,
            }
        }

    async def get_streaming_stats(self, credential: Optional[str] = None) -> Dict:
        """
        Get streaming usage statistics.
        
        Args:
            credential: Optional credential identifier to get stats for specific credential.
                       If None, returns aggregate stats for all credentials.
                       
        Returns:
            Dict containing streaming statistics
        """
        await self._ensure_initialized()
        
        async with self._data_lock:
            if credential:
                cred_id = mask_credential(credential)
                cred_data = self._usage_data.get("credentials", {}).get(cred_id, {})
                streaming_data = cred_data.get("streaming", self._create_default_credential_stats()["streaming"])
                streaming_data["active_streams"] = self._active_streams.get(cred_id, 0)
                return streaming_data
            else:
                # Aggregate stats
                total_starts = 0
                total_tokens = 0
                total_duration = 0
                total_active = sum(self._active_streams.values())
                
                for cred_id, cred_data in self._usage_data.get("credentials", {}).items():
                    streaming = cred_data.get("streaming", {})
                    total_starts += streaming.get("stream_starts", 0)
                    total_tokens += streaming.get("stream_tokens", 0)
                    total_duration += streaming.get("stream_duration_ms", 0)
                
                return {
                    "total_stream_starts": total_starts,
                    "total_stream_tokens": total_tokens,
                    "total_stream_duration_ms": total_duration,
                    "total_active_streams": total_active,
                    "credentials_with_streaming": len([
                        c for c in self._usage_data.get("credentials", {}).values() 
                        if c.get("streaming", {}).get("stream_starts", 0) > 0
                    ])
                }

    async def increment_usage(self, credential: str, error: bool = False) -> None:
        """
        Increment usage count for a credential.
        
        Args:
            credential: The credential identifier
            error: Whether this was an error response
        """
        await self._ensure_initialized()
        cred_id = mask_credential(credential)
        
        async with self._data_lock:
            if cred_id not in self._usage_data["credentials"]:
                self._usage_data["credentials"][cred_id] = self._create_default_credential_stats()
            
            self._usage_data["credentials"][cred_id]["usage_count"] += 1
            self._usage_data["credentials"][cred_id]["last_used"] = datetime.now(timezone.utc).isoformat()
            
            if error:
                self._usage_data["credentials"][cred_id]["error_count"] += 1
            
            await self._persist_usage_data()

    def _calculate_weight(self, credential: str, usage_count: int) -> float:
        """
        Calculate selection weight for a credential considering streaming load.
        
        The weight considers:
        - Base usage count
        - Active streaming connections (weighted by rotation_tolerance)
        - Provider rotation mode
        
        Args:
            credential: Credential identifier
            usage_count: Current usage count for the credential
            
        Returns:
            Weight value for random selection
        """
        cred_id = mask_credential(credential)
        
        # Get active streams for this credential
        active_streams = self._active_streams.get(cred_id, 0)
        
        # Effective usage includes active streams weighted by tolerance
        # Higher tolerance means active streams count less against the weight
        if self.rotation_tolerance > 0:
            # In random mode, active streams add to effective usage but scaled
            effective_usage = usage_count + (active_streams * (1.0 / (1.0 + self.rotation_tolerance)))
        else:
            # In deterministic mode, treat active streams as additional usage
            effective_usage = usage_count + active_streams
            
        # Get max usage across all credentials for normalization
        max_usage = max(
            (data.get("usage_count", 0) + self._active_streams.get(mask_credential(cred), 0) 
             for cred, data in self._usage_data.get("credentials", {}).items()),
            default=effective_usage
        )
        
        # Weight formula: lower usage = higher weight
        weight = (max_usage - effective_usage) + self.rotation_tolerance + 1
        
        return max(weight, 0.1)  # Ensure minimum weight

    async def get_credential_stats(self, credential: str) -> Dict:
        """
        Get complete statistics for a credential including streaming data.
        
        Args:
            credential: The credential identifier
            
        Returns:
            Dict with usage statistics
        """
        await self._ensure_initialized()
        cred_id = mask_credential(credential)
        
        async with self._data_lock:
            stats = self._usage_data.get("credentials", {}).get(cred_id, self._create_default_credential_stats())
            # Add current active streams from memory
            stats["active_streams"] = self._active_streams.get(cred_id, 0)
            return stats

    async def reset_daily_stats(self) -> None:
        """Reset daily statistics if needed based on configured reset time."""
        if not self.daily_reset_time_utc:
            return
            
        now = datetime.now(timezone.utc)
        current_time = now.time()
        
        last_reset_str = self._usage_data.get("last_reset")
        if last_reset_str:
            last_reset = datetime.fromisoformat(last_reset_str)
            
            # Check if we need to reset (crossed the reset time since last reset)
            if last_reset.date() < now.date() and current_time >= self.daily_reset_time_utc:
                lib_logger.info("Performing daily usage stats reset")
                
                async with self._data_lock:
                    # Archive current stats to daily history
                    date_key = last_reset.date().isoformat()
                    self._usage_data["daily_stats"][date_key] = {
                        "credentials": self._usage_data["credentials"].copy(),
                        "reset_time": last_reset.isoformat()
                    }
                    
                    # Reset current counters but preserve streaming totals
                    for cred_id in self._usage_data["credentials"]:
                        cred_data = self._usage_data["credentials"][cred_id]
                        cred_data["usage_count"] = 0
                        cred_data["error_count"] = 0
                        # Note: we preserve streaming totals as they are cumulative
                        
                    self._usage_data["last_reset"] = now.isoformat()
                    await self._persist_usage_data()

    async def cleanup_stale_streams(self, max_age_seconds: int = 300) -> int:
        """
        Clean up stale stream entries that may have not been properly closed.
        
        This should be called periodically to ensure active_streams count doesn't
        leak due to crashes or network failures.
        
        Args:
            max_age_seconds: Maximum age in seconds for a stream to be considered stale
            
        Returns:
            Number of stale streams cleaned up
        """
        cleaned = 0
        cutoff = datetime.now(timezone.utc).timestamp() - (max_age_seconds * 1000)
        
        async with self._stream_lock:
            async with self._data_lock:
                for cred_id in list(self._active_streams.keys()):
                    if self._active_streams[cred_id] > 0:
                        # Check last stream start time
                        cred_data = self._usage_data.get("credentials", {}).get(cred_id, {})
                        streaming = cred_data.get("streaming", {})
                        last_start = streaming.get("last_stream_start")
                        
                        if last_start:
                            last_start_ts = datetime.fromisoformat(last_start).timestamp()
                            if last_start_ts < cutoff:
                                lib_logger.warning(f"Cleaning up stale stream for {cred_id} (last start: {last_start})")
                                self._active_streams[cred_id] = 0
                                cleaned += 1
                                
        return cleaned

# Streaming Usage Tracking

## Overview

Phase 4.2 introduces comprehensive streaming usage tracking to the rotator library. This feature enables accurate load balancing and monitoring of long-running streaming connections to LLM providers.

## Features

- **Stream Lifecycle Tracking**: Record start and end of streaming sessions
- **Token Accounting**: Track total tokens processed through streams
- **Duration Metrics**: Measure cumulative streaming duration
- **Active Stream Management**: In-memory tracking of concurrent streams per credential
- **Stale Stream Cleanup**: Automatic cleanup of orphaned stream entries

## Usage

### Recording Stream Usage

```python
from rotator_library.usage_manager import UsageManager

usage_mgr = UsageManager()

# When starting a stream
await usage_mgr.record_stream_start(credential="path/to/cred.json")

# When ending a stream
await usage_mgr.record_stream_end(
    credential="path/to/cred.json", 
    tokens=150,  # Optional: tokens processed
    duration_ms=2500  # Optional: stream duration
)
```

### Retrieving Statistics

```python
# Get stats for specific credential
stats = await usage_mgr.get_streaming_stats(credential="path/to/cred.json")
# Returns:
# {
#     "stream_starts": 42,
#     "stream_tokens": 15000,
#     "stream_duration_ms": 45000,
#     "active_streams": 2,
#     "last_stream_start": "2024-01-15T10:30:00+00:00",
#     "last_stream_end": "2024-01-15T10:35:00+00:00"
# }

# Get aggregate stats
all_stats = await usage_mgr.get_streaming_stats()
```

### Load Balancing Integration

The UsageManager automatically incorporates active stream counts into the credential selection weight calculation:

- **Deterministic Mode** (`rotation_tolerance=0`): Active streams count as additional usage
- **Random Mode** (`rotation_tolerance>0`): Active streams are scaled by tolerance factor

This ensures credentials with many active streams are less likely to receive new requests, preventing overload.

### Data Persistence

Streaming statistics are persisted to the usage JSON file with the following schema:

```json
{
  "credentials": {
    "cred_id_hash": {
      "usage_count": 100,
      "streaming": {
        "stream_starts": 50,
        "stream_tokens": 10000,
        "stream_duration_ms": 300000,
        "last_stream_start": "2024-01-15T10:00:00+00:00",
        "last_stream_end": "2024-01-15T11:00:00+00:00"
      }
    }
  }
}
```

Note: Active stream counts are maintained in memory only and reset on restart. Stale stream cleanup is performed automatically to handle crashes.

## Configuration

No additional configuration is required. Streaming tracking is automatically available when using UsageManager.

## Migration

Existing usage data files are automatically migrated to version 2.0 format with streaming support on first load.
